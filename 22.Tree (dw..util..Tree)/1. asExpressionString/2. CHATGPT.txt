package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.ArrayList;
import java.util.List;

@RestController
public class TreeExpressionController {

    enum Kind { OBJECT_TYPE, ATTRIBUTE_TYPE }

    static class PathNode {
        Kind kind;
        String selector;

        PathNode(Kind kind, String selector) {
            this.kind = kind;
            this.selector = selector;
        }
    }

    @GetMapping("/asExpressionString")
    public String asExpressionString() {
        List<PathNode> nodes = new ArrayList<>();
        nodes.add(new PathNode(Kind.OBJECT_TYPE, "user"));
        nodes.add(new PathNode(Kind.ATTRIBUTE_TYPE, "name"));

        StringBuilder sb = new StringBuilder();
        for (PathNode node : nodes) {
            switch (node.kind) {
                case OBJECT_TYPE -> sb.append(".").append(node.selector);
                case ATTRIBUTE_TYPE -> sb.append(".@").append(node.selector);
            }
        }

        return sb.toString();
    }
}
