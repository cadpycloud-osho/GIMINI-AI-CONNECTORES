import java.util.*;

public class Main {

    static class TreeNode {
        String kind;
        String selector;
        String namespace;

        TreeNode(String kind, String selector, String namespace) {
            this.kind = kind;
            this.selector = selector;
            this.namespace = namespace;
        }
    }

    public static void main(String[] args) {

        Map<String, Boolean> result = new HashMap<>();

        // a
        List<TreeNode> pathA = Arrays.asList(
                new TreeNode("OBJECT_TYPE", "user", null),
                new TreeNode("ATTRIBUTE_TYPE", "name", null)
        );
        result.put("a", isAttributeType(pathA));

        // b
        List<TreeNode> pathB = Arrays.asList(
                new TreeNode("OBJECT_TYPE", "user", null),
                new TreeNode("ARRAY_TYPE", "name", null)
        );
        result.put("b", isAttributeType(pathB));

        // c
        List<TreeNode> pathC = Collections.singletonList(
                new TreeNode("ATTRIBUTE_TYPE", "name", null)
        );
        result.put("c", isAttributeType(pathC));

        System.out.println(result);
    }

    private static boolean isAttributeType(List<TreeNode> path) {
        if (path == null || path.isEmpty()) return false;
        TreeNode last = path.get(path.size() - 1);
        return "ATTRIBUTE_TYPE".equals(last.kind);
    }
}
