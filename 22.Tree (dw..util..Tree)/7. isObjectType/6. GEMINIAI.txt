package com.example.service;

import org.springframework.stereotype.Service;
import java.util.List;
import java.util.Map;
import java.util.LinkedHashMap;

@Service
public class TreeUtilService {

    /**
     * Replicates the logic of dw::util::Tree::isObjectType(path).
     * Checks the kind of the last AST node in the path list.
     * @param path The list representing the DataWeave AST path nodes.
     * @return true if the last node's kind is "OBJECT_TYPE", false otherwise.
     */
    @SuppressWarnings("unchecked")
    public boolean isObjectType(List<Map<String, Object>> path) {
        if (path == null || path.isEmpty()) {
            return false;
        }

        // Get the last node in the path list
        Map<String, Object> lastNode = path.get(path.size() - 1);

        // Check its 'kind' property
        return "OBJECT_TYPE".equals(lastNode.get("kind"));
    }

    /**
     * Executes the DataWeave simulation and returns the results.
     */
    public Map<String, Boolean> runIsObjectTypeSimulation() {
        // --- Input Structures ---

        // a: Last node is ATTRIBUTE_TYPE (Expected: false)
        List<Map<String, Object>> pathA = List.of(
            Map.of("kind", "OBJECT_TYPE", "selector", "user", "namespace", (Object) null),
            Map.of("kind", "ATTRIBUTE_TYPE", "selector", "name", "namespace", (Object) null)
        );

        // b: Last node is OBJECT_TYPE (Expected: true)
        List<Map<String, Object>> pathB = List.of(
            Map.of("kind", "OBJECT_TYPE", "selector", "user", "namespace", (Object) null),
            Map.of("kind", "OBJECT_TYPE", "selector", "name", "namespace", (Object) null)
        );

        // c: Last node is OBJECT_TYPE (Expected: true)
        List<Map<String, Object>> pathC = List.of(
            Map.of("kind", "OBJECT_TYPE", "selector", "user", "namespace", (Object) null)
        );

        // --- Execution ---
        Map<String, Boolean> output = new LinkedHashMap<>();
        output.put("a", isObjectType(pathA));
        output.put("b", isObjectType(pathB));
        output.put("c", isObjectType(pathC));

        return output;
    }
}
