import java.util.*;

public class Main {

    static class Node {
        String kind;
        Object selector;
        String namespace;

        Node(String kind, Object selector, String namespace) {
            this.kind = kind;
            this.selector = selector;
            this.namespace = namespace;
        }
    }

    public static void main(String[] args) {

        Map<String, Boolean> result = new HashMap<>();

        List<Node> pathA = Arrays.asList(
                new Node("OBJECT_TYPE", "user", null),
                new Node("ATTRIBUTE_TYPE", "name", null)
        );
        result.put("a", isArrayType(pathA));

        List<Node> pathB = Arrays.asList(
                new Node("OBJECT_TYPE", "user", null),
                new Node("ARRAY_TYPE", 0, null)
        );
        result.put("b", isArrayType(pathB));

        List<Node> pathC = Collections.singletonList(
                new Node("ARRAY_TYPE", 0, null)
        );
        result.put("c", isArrayType(pathC));

        System.out.println(result);
    }

    private static boolean isArrayType(List<Node> path) {
        for (Node node : path) {
            if ("ARRAY_TYPE".equals(node.kind)) {
                return true;
            }
        }
        return false;
    }
}
