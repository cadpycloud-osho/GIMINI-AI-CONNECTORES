@RequestMapping("/timer")
@RequiredArgsConstructor
class TimerController {
    
    private final TimerService timerService;
    
    @GetMapping("/currentMilliseconds")
    public Map<String, Long> currentMilliseconds() {
        return timerService.getCurrentMilliseconds();
    }
}

        return Map.of("currentMilliseconds", currentMilliseconds());
