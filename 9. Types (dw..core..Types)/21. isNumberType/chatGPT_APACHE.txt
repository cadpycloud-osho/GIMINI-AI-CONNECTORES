import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

public class CamelNumberTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type ANumber = Number (alias simulation)
                        Class<?> ANumber = Number.class;

                        // Simulating DataWeave isNumberType
                        boolean a = isNumberType(ANumber);
                        boolean b = isNumberType(Number.class);
                        boolean c = isNumberType(String.class);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave isNumberType
    private static boolean isNumberType(Class<?> clazz) {
        return Number.class.isAssignableFrom(clazz);
    }
}
