package com.example.typecheck.service;

import org.springframework.stereotype.Service;
import java.util.Map;

@Service
public class DateTypeCheckService {

    // Define the constant for the DataWeave 'Date' type representation
    private static final String DW_DATE_TYPE = "Date";

    /**
     * Helper method to simulate DataWeave's isDateType() function.
     * Checks if the given type name is equivalent to the 'Date' type.
     */
    private boolean isDateType(String typeName) {
        // Checks if the type name matches the expected "Date" type string.
        return typeName.equals(DW_DATE_TYPE);
    }

    /**
     * Converts the entire DataWeave script structure into a Map result.
     */
    public Map<String, Boolean> checkTypes() {
        
        // Emulating the type definitions and uses from the DW script:
        final String ADate_Type = "Date"; // Emulates type ADate = Date
        final String Builtin_Date = "Date";
        final String Builtin_String = "String";

        // Mimic the DataWeave transformation output structure:
        // { a: isDateType(ADate), b: isDateType(Date), c: isDateType(String) }
        return Map.of(
            "a", isDateType(ADate_Type),
            "b", isDateType(Builtin_Date),
            "c", isDateType(Builtin_String)
        );
    }
}
