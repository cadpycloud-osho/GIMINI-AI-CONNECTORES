import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

public class CamelRangeTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type ARange = Range (alias simulation)
                        Class<?> ARange = Range.class;

                        // Simulating DataWeave isRangeType
                        boolean a = isRangeType(ARange);
                        boolean b = isRangeType(Range.class);
                        boolean c = isRangeType(String.class);

                        String jsonOutput =

                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave isRangeType

        return clazz == Range.class;
    }

    }
