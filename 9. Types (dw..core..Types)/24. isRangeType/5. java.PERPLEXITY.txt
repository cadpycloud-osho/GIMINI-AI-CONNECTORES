import java.util.HashMap;
import java.util.Map;

class Range<T extends Comparable<T>> {
    private final T start;
    private final T end;
    
    public Range(T start, T end) {
        this.start = start;
        this.end = end;
    }
    
    public T getStart() { return start; }
    public T getEnd() { return end; }
}

public class Main {
    private static boolean isRangeType(Object value) {
        return value instanceof Range;
    }

    public static void main(String[] args) {
        Map<String, Boolean> result = new HashMap<>();

        Range<Integer> aRange = new Range<>(1, 10);   // ARange
        Range<Integer> rangeType = new Range<>(5, 15); // Range
        Object str = "test";                          // String

        result.put("a", isRangeType(aRange));
        result.put("b", isRangeType(rangeType));
        result.put("c", isRangeType(str));

        System.out.println(result);
    }
}
