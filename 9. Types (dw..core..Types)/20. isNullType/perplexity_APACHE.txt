public class DataWeaveNullTypeRoute extends RouteBuilder {

    public void configure() throws Exception {

            .to("log:output?showAll=true");

        Map<String, Boolean> result = new HashMap<>();

        // String - not Null
        result.put("c", isNullType(String.class));
        
        return result;
    }
    
    private boolean isNullType(Class<?> type) {
        // DataWeave Null type check equivalent
        // Returns true for Null and its aliases (explicit null representation)

               type.getName().contains("Null");
    }
