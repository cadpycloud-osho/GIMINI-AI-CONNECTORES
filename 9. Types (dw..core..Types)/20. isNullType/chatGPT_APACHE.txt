import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

public class CamelNullTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type ANull = Null (alias simulation)
                        Class<?> ANull = Null.class;

                        // Simulating DataWeave isNullType
                        boolean a = isNullType(ANull);
                        boolean b = isNullType(Null.class);
                        boolean c = isNullType(String.class);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave isNullType
    private static boolean isNullType(Class<?> clazz) {
        return clazz == Null.class;
    }

    // Simulating DataWeave Null type
    interface Null {
    }
}
