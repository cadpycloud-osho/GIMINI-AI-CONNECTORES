package com.example.demo;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.time.LocalTime;
import java.util.HashMap;
import java.util.Map;

@RestController
public class TypeCheckController {

    // Helper method for conceptual equivalence to isLocalTimeType(Type)
    private boolean isLocalTimeType(Class<?> type) {
        return LocalTime.class.isAssignableFrom(type);
    }

    @GetMapping("/typecheck")
    public Map<String, Boolean> checkTypes() {
        
        // DataWeave type ALocalTime is an alias for LocalTime
        Class<?> aLocalTime = LocalTime.class;
        
        // DataWeave type LocalTime
        Class<?> localTime = LocalTime.class;
        
        // DataWeave type String
        Class<?> stringType = String.class;

        // Perform the type checks
        boolean a = isLocalTimeType(aLocalTime);
        boolean b = isLocalTimeType(localTime);
        boolean c = isLocalTimeType(stringType);

        // Map the results to match the desired JSON structure
        Map<String, Boolean> output = new HashMap<>();
        output.put("a", a);
        output.put("b", b);
        output.put("c", c);
        
        return output; // Spring automatically converts this Map to JSON
    }
}
