package com.example.demo.controller;

import org.springframework.web.bind.annotation.RestController;

import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class LocalDateTimeTypeController {

    @GetMapping("/isLocalDateTimeType")
    public Map<String, Boolean> checkLocalDateTimeTypes() {

        Map<String, Boolean> result = new LinkedHashMap<>();

        // Equivalent to: type ALocalDateTime = LocalDateTime
        Class<?> ALocalDateTime = LocalDateTime.class;

        result.put("a", TypeUtils.isLocalDateTimeType(ALocalDateTime));
        result.put("b", TypeUtils.isLocalDateTimeType(LocalDateTime.class));
        result.put("c", TypeUtils.isLocalDateTimeType(String.class));

        return result;
    }
}
