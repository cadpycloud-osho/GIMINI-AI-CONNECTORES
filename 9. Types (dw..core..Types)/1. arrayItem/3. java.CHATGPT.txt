import java.util.LinkedHashMap;
import java.util.Map;

public class Main {

    public static String arrayItem(String type) {
        switch (type) {
            case "Array<String>":
                return "String";
            case "Array<Number>":
                return "Number";
            case "Array<Any>":
                return "Any";
            case "Array":
                return "Any";
            default:
                return "Unknown";
        }
    }

    public static void main(String[] args) {

        Map<String, String> result = new LinkedHashMap<>();
        result.put("a", arrayItem("Array<String>"));
        result.put("b", arrayItem("Array<Number>"));
        result.put("c", arrayItem("Array<Any>"));
        result.put("d", arrayItem("Array"));

        System.out.println(result);
    }
}
