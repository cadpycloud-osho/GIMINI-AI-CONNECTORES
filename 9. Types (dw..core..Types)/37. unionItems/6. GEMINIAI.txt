package com.example.demo;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

@RestController
public class UnionItemsController {

    // Define the constituent types of the AType Union
    private static final List<String> ATYPE_ITEMS = Arrays.asList("String", "Number");

    /**
     * Helper method to conceptually mimic unionItems(AType).
     */
    private List<String> unionItems() {
        // Returns the constant list of types that make up the union
        return ATYPE_ITEMS;
    }

    @GetMapping("/union-items")
    public Map<String, List<String>> getUnionItems() {
        
        // Perform the item extraction
        List<String> a = unionItems();

        // Map the results to match the desired JSON structure: { "a": [...] }
        Map<String, List<String>> output = new HashMap<>();
        output.put("a", a);
        
        return output; // Spring converts this Map structure directly to the output JSON
    }
}
