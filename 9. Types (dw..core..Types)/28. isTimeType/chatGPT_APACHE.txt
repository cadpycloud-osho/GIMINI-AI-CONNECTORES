import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

import java.time.LocalTime;

public class CamelTimeTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type ATime = Time (alias simulation)
                        Class<?> ATime = LocalTime.class;

                        // Simulating DataWeave isTimeType
                        boolean a = isTimeType(ATime);
                        boolean b = isTimeType(LocalTime.class);
                        boolean c = isTimeType(String.class);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave isTimeType
    private static boolean isTimeType(Class<?> clazz) {
        return clazz == LocalTime.class;
    }
}
