import java.util.HashMap;

public class DataWeaveTypeTypeRoute extends RouteBuilder {
    
    @Override
    public void configure() throws Exception {
        from("direct:start")

            .to("log:output?showAll=true");

    public Map<String, Boolean> processTypeTypeChecks() {
        Map<String, Boolean> result = new HashMap<>();

        // String - not Type
        result.put("c", isTypeType(String.class));
        
        return result;
    }
    
    private boolean isTypeType(Class<?> type) {
        // DataWeave Type type check equivalent (meta-type for all types)
        // Returns true for Type and its aliases
        return type == Type.class || 
               "Type".equals(type.getSimpleName()) ||
               type.getName().contains("Type") && 
               type.getName().endsWith("Type");
    }
