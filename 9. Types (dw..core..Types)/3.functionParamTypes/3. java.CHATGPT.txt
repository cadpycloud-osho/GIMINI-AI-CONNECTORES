import java.util.*;

public class Main {

    @FunctionalInterface
    interface AFunction {
        Number apply(String s, Number n);
    }

    @FunctionalInterface
    interface AFunction2 {
        Number apply();
    }

    public static List<Map<String, Object>> functionParamTypes(Class<?> clazz) {
        List<Map<String, Object>> list = new ArrayList<>();

        Method method = clazz.getMethods()[0];
        Class<?>[] params = method.getParameterTypes();

        for (Class<?> p : params) {
            Map<String, Object> map = new LinkedHashMap<>();
            map.put("paramType", p.getSimpleName());
            map.put("optional", false);
            list.add(map);
        }

        return list;
    }

    public static void main(String[] args) {

        Map<String, Object> result = new LinkedHashMap<>();

        result.put("a", functionParamTypes(AFunction.class));
        result.put("b", functionParamTypes(AFunction2.class));

        System.out.println(result);
    }
