import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

import java.time.Period;

public class CamelPeriodTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type APeriod = Period (alias simulation)
                        Class<?> APeriod = Period.class;

                        // Simulating DataWeave isPeriodType
                        boolean a = isPeriodType(APeriod);
                        boolean b = isPeriodType(Period.class);
                        boolean c = isPeriodType(String.class);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave isPeriodType
    private static boolean isPeriodType(Class<?> clazz) {
        return clazz == Period.class;
    }
}
