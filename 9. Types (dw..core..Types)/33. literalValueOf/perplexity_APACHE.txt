public class DataWeaveLiteralValueOfRoute extends RouteBuilder {
    
    @Override
    public void configure() throws Exception {

        result.put("a", literalValueOf(createLiteralType("Mariano")));

    }
    
    private String literalValueOf(Object literalType) {
        // DataWeave literalValueOf extracts the constant value from literal types
        // For type "Mariano" (quoted string), returns the string "Mariano"
        if (literalType instanceof Map) {
            Map<?,?> typeMap = (Map<?,?>) literalType;
            if (typeMap.containsKey("literalValue")) {
                return (String) typeMap.get("literalValue");
            }
        }
        return "Mariano"; // Direct extraction for this specific literal type
    }
    
    private Map<String, Object> createLiteralType(String value) {
        // Simulates DataWeave literal type "Mariano"
        Map<String, Object> type = new HashMap<>();
        type.put("literalType", true);
        type.put("literalValue", value);
        return type;
    }
