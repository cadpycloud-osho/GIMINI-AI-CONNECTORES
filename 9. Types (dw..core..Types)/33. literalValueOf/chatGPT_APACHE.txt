import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

public class CamelLiteralValueCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type AType = "Mariano"
                        String AType = "Mariano";

                        // Simulating DataWeave literalValueOf
                        String a = literalValueOf(AType);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": \"" + a + "\"\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave literalValueOf
    private static String literalValueOf(String literal) {
        return literal;
    }
