import java.util.HashMap;

public class Main {

    private static final String LITERAL_TYPE_INDICATOR = "\"";

    /**
     * Pure Java method to mimic the DataWeave isLiteralType() function.
     */
    public static boolean checkIsLiteralType(String typeRepresentation) {
        // Checks if the type representation starts and ends with a quote, indicating a string literal.
        return typeRepresentation.startsWith(LITERAL_TYPE_INDICATOR) && 
               typeRepresentation.endsWith(LITERAL_TYPE_INDICATOR);
    }
