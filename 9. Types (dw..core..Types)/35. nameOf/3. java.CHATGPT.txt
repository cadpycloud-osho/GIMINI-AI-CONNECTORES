import java.util.LinkedHashMap;
import java.util.Map;

public class Main {

    public static String nameOf(Class<?> type, String alias) {
        return alias != null ? alias : type.getSimpleName();
    }

    public static void main(String[] args) {

        Map<String, String> result = new LinkedHashMap<>();

        result.put("a", nameOf(String.class, "AArray"));
        result.put("b", nameOf(String.class, "AArray2"));
        result.put("c", nameOf(String.class, null));

        System.out.println(result);
    }
}
