import java.util.HashMap;

public class Main {

    private static final String ARRAY_TYPE_PREFIX = "Array<";
    
    /**
     * Pure Java method to mimic the DataWeave isArrayType() function.
     */
    public static boolean checkIsArrayType(String typeName) {
        // Checks if the type name starts with the array indicator
        return typeName.startsWith(ARRAY_TYPE_PREFIX);
    }

    public static void main(String[] args) {
        
        // 1. Define the input types (mimicking the DW type aliases and built-in types)

        boolean resultB = checkIsArrayType(Builtin_Boolean);

        // 3. Structure the final output map: { "a": resultA, "b": resultB }

        output.put("b", resultB);
