package com.example.demo;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.time.ZoneId; // The Java equivalent of DataWeave TimeZone
import java.lang.String;
import java.util.HashMap;
import java.util.Map;

@RestController
public class TimeZoneCheckController {

    /**
     * Helper method for conceptual equivalence to isTimeZoneType(Type).
     */
    private boolean isTimeZoneType(Class<?> type) {
        return ZoneId.class.isAssignableFrom(type);
    }

    @GetMapping("/timezone-typecheck")
    public Map<String, Boolean> checkTimeZoneTypes() {
        
        // DataWeave type ATimeZone is an alias for TimeZone
        Class<?> aTimeZone = ZoneId.class;
        
        // DataWeave type TimeZone
        Class<?> timeZoneType = ZoneId.class;
        
        // DataWeave type String
        Class<?> stringType = String.class;

        // Perform the type checks
        boolean a = isTimeZoneType(aTimeZone);
        boolean b = isTimeZoneType(timeZoneType);
        boolean c = isTimeZoneType(stringType);

        // Map the results to match the desired JSON structure
        Map<String, Boolean> output = new HashMap<>();
        output.put("a", a);
        output.put("b", b);
        output.put("c", c);
        
        return output; 
    }
}
