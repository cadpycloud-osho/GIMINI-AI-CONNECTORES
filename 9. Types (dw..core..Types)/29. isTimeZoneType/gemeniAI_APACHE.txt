public class TimeZoneTypeRoute extends RouteBuilder {

    public void configure() throws Exception {

                result.put("a", isTimeZoneType(refA));
                result.put("b", isTimeZoneType(refB));
                result.put("c", isTimeZoneType(refC));

                exchange.getIn().setBody(result);

            .marshal().json();

    private boolean isTimeZoneType(Object obj) {
        if (obj == null) return false;
        
        // Check if it's the class reference (ZoneId.class) or an actual instance
        // Note: java.util.TimeZone is also checked for compatibility
        return obj instanceof ZoneId || 
               obj instanceof java.util.TimeZone || 
               obj.equals(ZoneId.class);
    }
