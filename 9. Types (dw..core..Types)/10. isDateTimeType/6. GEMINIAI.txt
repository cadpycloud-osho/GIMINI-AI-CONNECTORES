package com.example.typecheck.service;

import org.springframework.stereotype.Service;
import java.util.Map;

@Service
public class DateTimeTypeCheckService {

    // Define the constant for the DataWeave 'DateTime' type representation
    private static final String DW_DATETIME_TYPE = "DateTime";

    /**
     * Helper method to simulate DataWeave's isDateTimeType() function.
     * Checks if the given type name is equivalent to the 'DateTime' type.
     */
    private boolean isDateTimeType(String typeName) {
        // Checks if the type name matches the expected "DateTime" type string.
        return typeName.equals(DW_DATETIME_TYPE);
    }

    /**
     * Converts the entire DataWeave script structure into a Map result.
     */
    public Map<String, Boolean> checkTypes() {
        
        // Emulating the type definitions and uses from the DW script:
        final String ADateTime_Type = "DateTime"; // Emulates type ADateTime = DateTime
        final String Builtin_DateTime = "DateTime";
        final String Builtin_String = "String";

        // Mimic the DataWeave transformation output structure:
        // { a: isDateTimeType(ADateTime), b: isDateTimeType(DateTime), c: isDateTimeType(String) }
        return Map.of(
            "a", isDateTimeType(ADateTime_Type),
            "b", isDateTimeType(Builtin_DateTime),
            "c", isDateTimeType(Builtin_String)
        );
    }
}
