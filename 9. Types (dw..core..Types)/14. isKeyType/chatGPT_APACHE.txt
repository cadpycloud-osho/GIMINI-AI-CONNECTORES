import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

public class CamelKeyTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // Simulating DataWeave isKeyType
                        boolean a = isKeyType(AKey.class);
                        boolean b = isKeyType(Key.class);
                        boolean c = isKeyType(Boolean.class);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave Key type check
    private static boolean isKeyType(Class<?> clazz) {
        return clazz == Key.class || Key.class.isAssignableFrom(clazz);
    }

    // Simulating DataWeave Key type
    interface Key {
    }

    // type AKey = Key
    interface AKey extends Key {
    }
}
