import java.util.HashMap;
import java.util.Map;
import java.util.regex.Pattern;

public class Main {

    private static boolean isRegexType(Object value) {
        return value instanceof Pattern;
    }

    public static void main(String[] args) {
        Map<String, Boolean> result = new HashMap<>();
        
        Pattern aRegex = Pattern.compile(".*");     // ARegex
        Pattern regexType = Pattern.compile("\\d+"); // Regex
        Boolean boolValue = true;                   // Boolean
        
        result.put("a", isRegexType(aRegex));
        result.put("b", isRegexType(regexType));
        result.put("c", isRegexType(boolValue));
        
        System.out.println(result);
    }
}

{a=true, b=true, c=false}
