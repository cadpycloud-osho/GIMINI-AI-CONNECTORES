import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

import java.util.regex.Pattern;

public class CamelRegexTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type ARegex = Regex (alias simulation)
                        Class<?> ARegex = Pattern.class;

                        // Simulating DataWeave isRegexType
                        boolean a = isRegexType(ARegex);
                        boolean b = isRegexType(Pattern.class);
                        boolean c = isRegexType(Boolean.class);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave isRegexType
    private static boolean isRegexType(Class<?> clazz) {
        return clazz == Pattern.class;
    }
}
