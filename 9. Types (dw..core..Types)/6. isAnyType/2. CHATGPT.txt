package com.example.demo.util;

public class TypeUtils {

    public static boolean isAnyType(String typeName) {
        // DW Behavior:
        // Any or alias of Any must return true
        if (typeName == null) return false;

        return typeName.equalsIgnoreCase("Any")
                || typeName.equalsIgnoreCase("AAny");
    }
}

package com.example.demo.controller;

import com.example.demo.util.TypeUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class TypeCheckController {

    @GetMapping("/isanytype")
    public Map<String, Boolean> checkTypes() {

        Map<String, Boolean> result = new LinkedHashMap<>();

        result.put("a", TypeUtils.isAnyType("AAny"));
        result.put("b", TypeUtils.isAnyType("Any"));
        result.put("c", TypeUtils.isAnyType("String"));

        return result;
    }
}
