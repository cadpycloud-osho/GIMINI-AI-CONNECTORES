import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;

import java.net.URI;

public class CamelUriTypeCheck {

    public static void main(String[] args) throws Exception {

        DefaultCamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {

                from("direct:start")
                    .process(exchange -> {

                        // type AUri = Uri (alias simulation)
                        Class<?> AUri = URI.class;

                        // Simulating DataWeave isUriType
                        boolean a = isUriType(AUri);
                        boolean b = isUriType(URI.class);
                        boolean c = isUriType(String.class);

                        String jsonOutput =
                                "{\n" +
                                "  \"a\": " + a + ",\n" +
                                "  \"b\": " + b + ",\n" +
                                "  \"c\": " + c + "\n" +
                                "}";

                        exchange.getMessage().setBody(jsonOutput);
                    })
                    .to("log:output");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(2000);
        context.stop();
    }

    // Helper method to simulate DataWeave isUriType
    private static boolean isUriType(Class<?> clazz) {
        return clazz == URI.class;
    }
}
