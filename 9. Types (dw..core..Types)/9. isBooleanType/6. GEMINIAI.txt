package com.example.typecheck.service;

import org.springframework.stereotype.Service;
import java.util.Map;

@Service
public class BooleanTypeCheckService {

    // Define the constant for the DataWeave 'Boolean' type representation
    private static final String DW_BOOLEAN_TYPE = "Boolean";

    /**
     * Helper method to simulate DataWeave's isBooleanType() function.
     * Checks if the given type name is equivalent to the 'Boolean' type.
     */
    private boolean isBooleanType(String typeName) {
        // Checks if the type name matches the expected "Boolean" type string.
        return typeName.equals(DW_BOOLEAN_TYPE);
    }

    /**
     * Converts the entire DataWeave script structure into a Map result.
     */
    public Map<String, Boolean> checkTypes() {
        
        // Emulating the type definitions and uses from the DW script:
        final String ABoolean_Type = "Boolean"; // Emulates type ABoolean = Boolean
        final String Builtin_Boolean = "Boolean";
        final String Builtin_String = "String";

        // Mimic the DataWeave transformation output structure:
        // { a: isBooleanType(ABoolean), b: isBooleanType(Boolean), c: isBooleanType(String) }
        return Map.of(
            "a", isBooleanType(ABoolean_Type),
            "b", isBooleanType(Builtin_Boolean),
            "c", isBooleanType(Builtin_String)
        );
    }
}
