import java.util.HashMap;
import java.util.Map;

public class Main {

    static class DataFormatDescriptor {
        String name;
        String defaultMimeType;

        DataFormatDescriptor(String name, String defaultMimeType) {
            this.name = name;
            this.defaultMimeType = defaultMimeType;
        }
    }

    public static void main(String[] args) {

        DataFormatDescriptor jsonDF = findDataFormatDescriptorByMime("json");
        DataFormatDescriptor unknownDF = findDataFormatDescriptorByMime(null);

        Map<String, Map<String, String>> result = new HashMap<>();
        result.put("json", simplify(jsonDF));
        result.put("unknown", simplify(unknownDF));

        System.out.println(result);
    }

    private static DataFormatDescriptor findDataFormatDescriptorByMime(String subtype) {
        if ("json".equals(subtype)) {
            return new DataFormatDescriptor("json", "application/json");
        }
        return null; // unknown
    }

    private static Map<String, String> simplify(DataFormatDescriptor df) {
        Map<String, String> simplified = new HashMap<>();
        if (df != null) {
            simplified.put("name", df.name);
            simplified.put("defaultMimeType", df.defaultMimeType);
        } else {
            simplified.put("name", "unknown");
            simplified.put("defaultMimeType", "unknown");
        }
        return simplified;
    }
}
