import org.springframework.core.env.Environment;
import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

@RestController
public class PropsController {

    private final Environment env;

    public PropsController(Environment env) {
        this.env = env;
    }

    @GetMapping(value = "/props", produces = MediaType.APPLICATION_JSON_VALUE)
    public Map<String, String> getProps() {
        // Matches: { "props" : props() }
        Map<String, String> props = new HashMap<>();
        
        // System properties (matches Mule runtime)
        System.getProperties().forEach((key, value) -> 
            props.put(key.toString(), value.toString()));
        
        // Spring environment properties
        env.getPropertySources().forEach(ps -> {
            if (ps.getSource() instanceof Map) {
                @SuppressWarnings("unchecked")
                Map<String, Object> source = (Map<String, Object>) ps.getSource();
                source.forEach((key, value) -> 
                    props.putIfAbsent(key, value.toString()));
            }
        });
        
        return Map.of("props", String.valueOf(props));
    }
}
