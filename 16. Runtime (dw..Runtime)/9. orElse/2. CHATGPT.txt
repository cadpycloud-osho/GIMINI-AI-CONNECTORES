package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

@RestController
public class TryOrElseController {

    @GetMapping("/tryOrElse")
    public Map<String, String> getUserNames() {

        Map<String, Object> user = new HashMap<>(); // {}
        Map<String, Object> otherUser = new HashMap<>();
        otherUser.put("name", "DW");

        Map<String, String> result = new HashMap<>();

        // Equivalent of DW: try(() -> user.name!) orElse "No User Name"
        try {
            result.put("a", (String) user.get("name")); // will return null
            if (result.get("a") == null) throw new NullPointerException();
        } catch (Exception e) {
            result.put("a", "No User Name");
        }

        try {
            result.put("b", (String) otherUser.get("name")); // has value "DW"
            if (result.get("b") == null) throw new NullPointerException();
        } catch (Exception e) {
            result.put("b", "No User Name");
        }

        return result;
    }
}
