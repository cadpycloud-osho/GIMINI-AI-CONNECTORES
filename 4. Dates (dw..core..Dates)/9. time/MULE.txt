time(parts: TimeFactory): Time
Creates a Time value from values specified for hour, minutes, seconds, and timezone fields.




Source
%dw 2.0
import * from dw::core::Dates

{
  newTime: time({ hour: 12, minutes: 30, seconds: 40 , timeZone: |-03:00| })
}
