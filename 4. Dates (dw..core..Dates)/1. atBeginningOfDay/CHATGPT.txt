atBeginningOfDay:


example 1:


input:


@GetMapping("/beginningOfDay1")
public Map<String, String> atBeginningOfDayExample1() {
    OffsetDateTime dateTime = OffsetDateTime.parse("2020-10-06T18:23:20.351-03:00");

    OffsetDateTime startOfDay = dateTime
            .toLocalDate()
            .atStartOfDay()
            .atOffset(dateTime.getOffset());

    return Map.of("atBeginningOfDayDateTime", startOfDay.toString());
}


output:

{
  "atBeginningOfDayDateTime": "2020-10-06T00:00:00-03:00"
}

example 2:


input:


@GetMapping("/beginningOfDay2")
public Map<String, String> atBeginningOfDayExample2() {
    LocalDateTime dateTime = LocalDateTime.parse("2020-10-06T18:23:20.351");

    LocalDateTime startOfDay = dateTime.toLocalDate().atStartOfDay();

    return Map.of("atBeginningOfDayLocalDateTime", startOfDay.toString());
}


output:

{
  "atBeginningOfDayLocalDateTime": "2020-10-06T00:00:00"
}
