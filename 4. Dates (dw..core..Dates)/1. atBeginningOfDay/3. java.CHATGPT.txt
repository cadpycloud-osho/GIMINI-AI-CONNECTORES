atBeginningOfDay:

example 1:


input:

import java.time.OffsetDateTime;
import java.time.ZoneOffset;

public class Main {

    public static void main(String[] args) {

        OffsetDateTime input =
                OffsetDateTime.parse("2020-10-06T18:23:20.351-03:00");

        OffsetDateTime result =
                input.toLocalDate()
                     .atStartOfDay()
                     .atOffset(input.getOffset());

        System.out.println(result);
    }
}


output:

2020-10-06T00:00-03:00
