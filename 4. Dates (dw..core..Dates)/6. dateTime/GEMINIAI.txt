import java.time.LocalDateTime;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;

public class DateTimeCreationExample {

    /**
     * DataWeave equivalent: dateTime({year: ..., timeZone: ...})
     * Creates a ZonedDateTime object from component fields and a timezone offset.
     * * @param year The year component (e.g., 2012).
     * @param month The month component (1 to 12).
     * @param day The day component (1 to 31).
     * @param hour The hour component (0 to 23).
     * @param minute The minute component (0 to 59).
     * @param second The second component (0 to 59).
     * @param offsetId The timezone offset string (e.g., "-03:00").
     * @return The resulting ZonedDateTime object.
     */
    public static ZonedDateTime createDateTime(int year, int month, int day, int hour, int minute, int second, String offsetId) {
        // 1. Create the LocalDateTime component (Date and Time without offset)
        LocalDateTime localDateTime = LocalDateTime.of(year, month, day, hour, minute, second);
        
        // 2. Create the ZoneOffset component
        ZoneOffset offset = ZoneOffset.of(offsetId);
        
        // 3. Combine them to create the ZonedDateTime
        return ZonedDateTime.of(localDateTime, offset);
    }

    public static void main(String[] args) {
        // 1. Define the input parameters from DataWeave
        int year = 2012;
        int month = 10;
        int day = 11;
        int hour = 12;
        int minute = 30;
        int second = 40;
        String timeZoneOffset = "-03:00"; // |-03:00| in DW

        System.out.println("--- DateTime Creation Example ---");
        
        // 2. Call the Java function
        ZonedDateTime newDateTime = createDateTime(year, month, day, hour, minute, second, timeZoneOffset);
        
        // 3. Format the output to match the DataWeave ISO 8601 format
        String formattedOutput = newDateTime.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME);
        
        System.out.println("Source Input (Components): {year: 2012, ..., timeZone: " + timeZoneOffset + "}");
        System.out.println("Result Output: " + formattedOutput);
    }
}
