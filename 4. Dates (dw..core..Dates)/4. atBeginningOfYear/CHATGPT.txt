


atBeginningOfYear:



example 1:


input:

@GetMapping("/beginningOfYear1")
public Map<String, String> atBeginningOfYearExample1() {
    OffsetDateTime dateTime = OffsetDateTime.parse("2020-10-06T18:23:20.351-03:00");

    OffsetDateTime startOfYear = dateTime
            .withDayOfYear(1)
            .toLocalDate()
            .atStartOfDay()
            .atOffset(dateTime.getOffset());

    return Map.of("atBeginningOfYearDateTime", startOfYear.toString());
}


output:



{
  "atBeginningOfYearDateTime": "2020-01-01T00:00:00-03:00"
}

@GetMapping("/beginningOfYear2")
