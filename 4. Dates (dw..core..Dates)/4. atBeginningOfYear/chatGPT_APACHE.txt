atBeginningOfYear:

example 1:


input:

from("direct:atBeginningOfYearDateTime")
.process(exchange -> {

    // Input DateTime with ZoneOffset
    java.time.OffsetDateTime input =
            java.time.OffsetDateTime.parse("2020-10-06T18:23:20.351-03:00");

    // Beginning of the year (set month/day to Jan 1, time to 00:00:00.000)
    java.time.OffsetDateTime result =
            input.withDayOfYear(1)
                 .withHour(0)
                 .withMinute(0)
                 .withSecond(0)
                 .withNano(0);

    // JSON-style output
    java.util.Map<String, Object> output = new java.util.HashMap<>();
    output.put("atBeginningOfYearDateTime", result.toString());

    exchange.getIn().setBody(output);
});


output:

{
  "atBeginningOfYearDateTime": "2020-01-01T00:00:00-03:00"
}
