import java.time.LocalDate;
import java.time.format.DateTimeFormatter;

public class DateCreationExample {

    /**
     * DataWeave equivalent: date({year: 2012, month: 10, day: 11})
     * Creates a LocalDate object from component fields.
     * * @param year The year component (e.g., 2012).
     * @param month The month component (1 to 12).
     * @param day The day component (1 to 31).
     * @return The resulting LocalDate object.
     */
    public static LocalDate createDate(int year, int month, int day) {
        // Use the static factory method LocalDate.of()
        return LocalDate.of(year, month, day);
    }

    public static void main(String[] args) {
        // 1. Define the input parameters from DataWeave
        int year = 2012;
        int month = 10;
        int day = 11;
        
        System.out.println("--- Date Creation Example ---");
        
        // 2. Call the Java function
        LocalDate newDate = createDate(year, month, day);
        
        // 3. Format the output to match the DataWeave ISO 8601 format
        String formattedOutput = newDate.format(DateTimeFormatter.ISO_LOCAL_DATE);
        
        System.out.println("Source Input (Components): {year: " + year + ", month: " + month + ", day: " + day + "}");
        System.out.println("Result Output: " + formattedOutput);
    }
}
