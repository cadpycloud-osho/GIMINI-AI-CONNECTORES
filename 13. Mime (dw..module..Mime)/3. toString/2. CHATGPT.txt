@RestController
public class MimeExample1Controller {

    @GetMapping("/mime1")
    public Map<String, String> mimeExample1() {

        String type = "application";
        String subtype = "json";
        Map<String, String> parameters = new HashMap<>();

        String result = buildMime(type, subtype, parameters);

        Map<String, String> response = new HashMap<>();
        response.put("mime", result);
        return response;
    }

    private String buildMime(String type, String subtype, Map<String, String> params) {
        StringBuilder sb = new StringBuilder();
        sb.append(type).append("/").append(subtype);
        if (!params.isEmpty()) {
            params.forEach((k, v) -> sb.append(";").append(k).append("=").append(v));
        }
        return sb.toString();
    }
}

output:


{ "mime": "application/json" }
