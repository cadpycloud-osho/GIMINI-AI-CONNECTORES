import java.util.Map;
import java.util.stream.Collectors;

public class Main {

    // 1. Conceptual Class to mimic the structure of the input (MimeResult)
    record MimeResult(String type, String subtype, Map<String, String> parameters) {}

    /**
     * Conceptual method to emulate the composition of a Content-Type string 
     * from a type, subtype, and a map of parameters.
     */
    public static String convertToContentType(String type, String subtype, Map<String, String> parameters) {
        
        // 1. Start with the base MIME type: type/subtype
        StringBuilder mimeBuilder = new StringBuilder();
        mimeBuilder.append(type).append("/").append(subtype);

        // 2. Append parameters if they exist
        if (parameters != null && !parameters.isEmpty()) {
            
            // Map the parameters into the format "; key=value"
            String parameterString = parameters.entrySet().stream()
                .map(entry -> entry.getKey() + "=" + entry.getValue())
                .collect(Collectors.joining("; ", "; ", ""));
            
            mimeBuilder.append(parameterString);
        }
        
        return mimeBuilder.toString();
    }

    public static void main(String[] args) {
        
        // Input MimeResult object
        MimeResult withParam = new MimeResult("text", "plain", 
            Map.of("charset", "UTF-8"));

        // Perform the conceptual conversion
        String result = convertToContentType(withParam.type(), withParam.subtype(), withParam.parameters());

        // Output the result in the requested JSON format
        System.out.println("{");
        System.out.println("  \"result\": \"" + result + "\"");
        System.out.println("}");
    }
}

  "result": "text/plain; charset=UTF-8"
