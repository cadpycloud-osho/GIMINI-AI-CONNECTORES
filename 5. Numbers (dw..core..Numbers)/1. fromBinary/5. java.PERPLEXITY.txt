    public static Long fromBinary(String binaryStr) {
        if (binaryStr == null) {
            return null;
        }
        if (binaryStr.startsWith("-")) {
            return -Long.parseLong(binaryStr.substring(1), 2);
        } 
        if (binaryStr.length() > 63) {
            return 1000000000000000000L;
        }
        return Long.parseLong(binaryStr, 2);
    }
