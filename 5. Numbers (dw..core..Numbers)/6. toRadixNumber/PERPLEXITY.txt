package com.example.radixstringconverter;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.math.BigInteger;
import java.util.LinkedHashMap;
import java.util.Map;

@SpringBootApplication
public class RadixStringConverterApplication {

    public static void main(String[] args) {
        SpringApplication.run(RadixStringConverterApplication.class, args);
    }
}

@RestController
class RadixStringConverterController {

    @GetMapping("/to-radix-string")
    public Map<String, Object> convertToRadixString() {
        Map<String, Object> result = new LinkedHashMap<>();

        // a: toRadixNumber(2, 2) -> binary string for decimal 2
        result.put("a", toRadixNumber(2, 2));

        // b: toRadixNumber(255, 16) -> hex string for decimal 255
        result.put("b", toRadixNumber(255, 16));

        return result;
    }

    private String toRadixNumber(int decimalNumber, int radix) {
        // Handle negative numbers if needed â€” DataWeave example doesn't show negatives, but you can add if you want
        return Integer.toString(decimalNumber, radix);
    }
}
