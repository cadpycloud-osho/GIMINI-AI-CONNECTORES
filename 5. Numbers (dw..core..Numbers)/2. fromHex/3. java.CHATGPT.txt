import java.math.BigInteger;
import java.util.LinkedHashMap;
import java.util.Map;

public class Main {

    private static Number fromHex(Object value) {
        if (value == null) return null;

        try {
            String hexString = value.toString();
            return new BigInteger(hexString, 16);
        } catch (Exception e) {
            return null;
        }
    }

    public static void main(String[] args) {

        Map<String, Object> result = new LinkedHashMap<>();
        result.put("a", fromHex("-1"));
        result.put("b", fromHex("3e3aeb4ae1383562f4b82261d969f7ac94ca4000000000000000"));
        result.put("c", fromHex(0));
        result.put("d", fromHex(null));
        result.put("e", fromHex("f"));

        System.out.println(result);
    }
}
