    String d = null;

    java.util.Map<String, Object> output = new java.util.HashMap<>();

    // Convert hex strings to decimal
    output.put("a", a != null ? new java.math.BigInteger(a, 16) : null);
    output.put("b", b != null ? new java.math.BigInteger(b, 16) : null);
    output.put("c", c != null ? new java.math.BigInteger(c, 16) : null);
    output.put("d", d != null ? new java.math.BigInteger(d, 16) : null);
    output.put("e", e != null ? new java.math.BigInteger(e, 16) : null);

    exchange.getIn().setBody(output);
});
