    // DataWeave Numbers::fromHex equivalent
    public static BigInteger fromHex(String hex) {
        if (hex == null) {
            return null;
        }
        if (hex instanceof String && ((String) hex).startsWith("-")) {
            String positivePart = hex.substring(1);
            return new BigInteger(positivePart, 16).negate();
        } else {
            return new BigInteger(hex, 16);
        }
    }
