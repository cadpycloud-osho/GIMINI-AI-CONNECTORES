package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

@RestController
public class MathAcosController {

    @GetMapping("/acos")
    public Map<String, Double> acosValues() {

        Map<String, Double> result = new HashMap<>();

        result.put("acos0", acos(0));
        result.put("acos13", acos(0.13));
        result.put("acos-1", acos(-1));
        result.put("acos1", acos(1));
        result.put("acos1.1", acos(1.1));

        return result;
    }

    // Equivalent of DW acos()
    private Double acos(double value) {
        double result = Math.acos(value);
        return Double.isNaN(result) ? null : result;
    }
}
