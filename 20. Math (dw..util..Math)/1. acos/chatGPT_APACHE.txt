import java.util.*;

public class AcosExample {

    public static void main(String[] args) {

        Map<String, Object> output = new LinkedHashMap<>();

        output.put("acos0", safeAcos(0.0));
        output.put("acos13", safeAcos(0.13));
        output.put("acos-1", safeAcos(-1.0));
        output.put("acos1", safeAcos(1.0));
        output.put("acos1.1", safeAcos(1.1)); // outside valid range [-1,1]

        System.out.println(output);
    }

    private static Double safeAcos(double value) {
        // Math.acos returns NaN for invalid values, we convert it to null
        double result = Math.acos(value);
        if (Double.isNaN(result)) {
            return null;
        }
        return result;
    }
}
