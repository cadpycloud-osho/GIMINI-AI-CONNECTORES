import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.math.BigDecimal;
import java.util.HashMap;
import java.util.Map;

@RestController
public class MathController {

    @GetMapping("/toDegrees")
    public Map<String, BigDecimal> toDegreesExample() {

        Map<String, BigDecimal> result = new HashMap<>();

        result.put("toDegrees0.17", BigDecimal.valueOf(Math.toDegrees(0.174)));
        result.put("toDegrees0", BigDecimal.valueOf(Math.toDegrees(0)));
        result.put("toDegrees-20", BigDecimal.valueOf(Math.toDegrees(-0.20)));

        return result;
    }
}
