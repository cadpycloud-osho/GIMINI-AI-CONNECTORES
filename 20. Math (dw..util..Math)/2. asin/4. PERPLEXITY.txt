@RestController
@RequestMapping("/math/asin")
@RequiredArgsConstructor
class AsinController {
    
    private final MathService mathService;
    
    @GetMapping("/test")
    public Map<String, Double> testAsin() {
        return mathService.executeAsinTests();
    }
}

@Service
class MathService {
    
    public Map<String, Double> executeAsinTests() {
        Map<String, Double> result = new HashMap<>();
        
        result.put("asin0", asin(0.0));
        result.put("asin13", asin(0.13));
        result.put("asin-1", asin(-1.0));
        result.put("asin1.1", asin(1.1));
        
        return result;
    }
    
    /**
     * DataWeave asin() equivalent:
     * Returns arcsine of x, or null if x is outside [-1, 1] range
     */
    public Double asin(double x) {
        // Check domain: x must be in [-1, 1]
        if (x < -1.0 || x > 1.0) {
            return null;
        }
        return Math.asin(x);
    }
}
