@Service
class MathService {
    
    public Map<String, Double> executeTanTests() {
        Map<String, Double> result = new HashMap<>();
        
        result.put("tan0", tan(0.0));
        result.put("tan13", tan(0.13));
        result.put("tan-1", tan(-1.0));
        
        return result;
    }
    
    /**
     * DataWeave tan() equivalent:
     * Returns tangent of x (in radians)
     */
    public Double tan(double x) {
        return Math.tan(x);
    }
}
