@RestController
@RequestMapping("/math/cos")
@RequiredArgsConstructor
class CosController {
    
    private final MathService mathService;
    
    @GetMapping("/test")
    public Map<String, Double> testCos() {
        return mathService.executeCosTests();
    }
}

@Service
class MathService {
    
    public Map<String, Double> executeCosTests() {
        Map<String, Double> result = new HashMap<>();
        
        result.put("cos0", cos(0.0));
        result.put("cos13", cos(0.13));
        result.put("cos-1", cos(-1.0));
        
        return result;
    }
    
    /**
     * DataWeave cos() equivalent:
     * Returns cosine of x (in radians)
     */
    public Double cos(double x) {
        return Math.cos(x);
    }
}
