package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

@RestController
public class MathLogController {

    @GetMapping("/log10")
    public Map<String, Double> getLog10Values() {

        Map<String, Double> result = new HashMap<>();

        result.put("log1010", log10(10));
        result.put("log1013", log10(0.13));
        result.put("log10-20", log10(-20));

        return result;
    }

    // Equivalent of dw::util::Math::log10
    private Double log10(double value) {
        if (value <= 0) {
            return null; // DataWeave behavior
        }
        return Math.log10(value);
    }
}
