import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.util.UriComponentsBuilder;
import java.util.HashMap;
import java.util.Map;

@RestController
public class UrlComposeController {

    @GetMapping("/url-compose")
    public Map<String, String> composeUrl() {
        Map<String, String> result = new HashMap<>();
        String urlPath = "content folder";
        
        // Replicates both compose(["http://examplewebsite.com/", "/page.html"], ["$(urlPath)"])
        // and compose `http://examplewebsite.com/$(urlPath)/page.html`
        String encodedURL = UriComponentsBuilder
            .fromHttpUrl("http://examplewebsite.com")
            .path(urlPath)
            .path("/page.html")
            .build()
            .toUriString();
        
        result.put("encodedURL", encodedURL);
        return result;
    }
}
