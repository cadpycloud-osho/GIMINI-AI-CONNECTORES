    /**
     * Conceptually mimics DataWeave's URL::decodeURI.
     * Performs URL percent-decoding on the input string using UTF-8.
     * @param encodedURI The URI string containing percent-encoded characters.
     * @return The decoded string.
     */
    public static String decodeUri(String encodedURI) {
        try {
            // Use URLDecoder.decode with the standard UTF-8 charset
            return URLDecoder.decode(encodedURI, StandardCharsets.UTF_8);
        } catch (Exception e) {
            // Handle exceptions like IllegalArgumentException if the encoding is malformed
            return "Decoding Error: " + e.getMessage();
        }
    }

    public static void main(String[] args) {
