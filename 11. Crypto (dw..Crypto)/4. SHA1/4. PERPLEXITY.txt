import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.security.MessageDigest;
import java.util.HashMap;
import java.util.Map;

@RestController
public class SHA1Controller {

    @GetMapping("/sha1")
    public Map<String, String> computeSHA1() {
        String input = "dsasd";
        String sha1Hex = computeSHA1Hex(input);
        
        Map<String, String> response = new HashMap<>();
        response.put("sha1", sha1Hex);
        return response;
    }

    private String computeSHA1Hex(String input) {
        try {
            MessageDigest md = MessageDigest.getInstance("SHA-1");
            byte[] digest = md.digest(input.getBytes());
            return bytesToHex(digest);
        } catch (Exception e) {
            throw new RuntimeException("SHA1 computation failed", e);
        }
    }

    private String bytesToHex(byte[] bytes) {
        StringBuilder sb = new StringBuilder(bytes.length * 2);
        for (byte b : bytes) {
            sb.append(String.format("%02x", b));
        }
        return sb.toString();
    }
}
