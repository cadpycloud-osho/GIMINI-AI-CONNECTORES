public class Main {

    // Generic someEntry method
    private static boolean someEntry(Map<String, Object> map, EntryPredicate predicate) {
        if (map == null) return false;
        for (Map.Entry<String, Object> entry : map.entrySet()) {
            if (predicate.test(entry.getValue(), entry.getKey())) {
                return true;
            }
        }
        return false;
    }
