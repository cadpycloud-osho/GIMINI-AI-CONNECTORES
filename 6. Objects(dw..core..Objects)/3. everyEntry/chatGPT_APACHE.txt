import org.apache.camel.Exchange;
import org.apache.camel.Processor;
import java.util.Map;
import java.util.HashMap;

public class EveryEntryProcessor implements Processor {

    @Override
    public void process(Exchange exchange) throws Exception {
        // Example input maps
        Map<String, Object> a = new HashMap<>();
        Map<String, Object> b = new HashMap<>();
        b.put("a", "");
        b.put("b", "123");
        Map<String, Object> c = new HashMap<>();
        c.put("a", "");
        c.put("b", 123);
        Map<String, Object> d = new HashMap<>();
        d.put("a", "");
        d.put("b", 123);
        Map<String, Object> e = new HashMap<>();
        e.put("a", "");

        Map<String, Boolean> results = new HashMap<>();

        // a: {} everyEntry (value is String)
        results.put("a", a.entrySet().stream().allMatch(entry -> entry.getValue() instanceof String));

        // b: {a:"", b:"123"} everyEntry (value is String)
        results.put("b", b.entrySet().stream().allMatch(entry -> entry.getValue() instanceof String));

        // c: {a:"", b:123} everyEntry (value is String)
        results.put("c", c.entrySet().stream().allMatch(entry -> entry.getValue() instanceof String));

        // d: {a:"", b:123} everyEntry (key == "a")
        results.put("d", d.entrySet().stream().allMatch(entry -> entry.getKey().equals("a")));

        // e: {a:""} everyEntry (key == "a")
        results.put("e", e.entrySet().stream().allMatch(entry -> entry.getKey().equals("a")));

        // f: null map
        Map<String, Object> f = null;
        results.put("f", f == null || f.entrySet().stream().allMatch(entry -> entry.getKey().equals("a")));

        // Set the result in the exchange
        exchange.getIn().setBody(results);
    }
}

from("direct:start")
    .process(new EveryEntryProcessor())
    .to("log:results");
