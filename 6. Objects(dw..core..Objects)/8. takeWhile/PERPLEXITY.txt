package com.example.takewhile;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.LinkedHashMap;
import java.util.Map;

@SpringBootApplication
public class TakeWhileApplication {

    public static void main(String[] args) {
        SpringApplication.run(TakeWhileApplication.class, args);
    }
}

@RestController
class TakeWhileController {

    @GetMapping("/take-while")
    public Map<String, Object> takeWhileExample() {
        Map<String, Integer> obj = new LinkedHashMap<>();
        obj.put("a", 1);
        obj.put("b", 2);
        obj.put("c", 5);
        obj.put("d", 1);

        Map<String, Integer> result = takeWhile(obj, (value, key) -> value < 3);

        return Map.of("result", result);
    }

    /**
     * Returns entries from the start of the map while predicate(value, key) is true.
     * Stops at first entry where predicate is false.
     */
    private <K, V> Map<K, V> takeWhile(Map<K, V> map, EntryPredicate<V, K> predicate) {
        Map<K, V> result = new LinkedHashMap<>();
        for (Map.Entry<K, V> entry : map.entrySet()) {
            if (predicate.test(entry.getValue(), entry.getKey())) {
                result.put(entry.getKey(), entry.getValue());
            } else {
                break;
            }
        }
        return result;
    }

    @FunctionalInterface
    interface EntryPredicate<V, K> {
        boolean test(V value, K key);
    }
}
