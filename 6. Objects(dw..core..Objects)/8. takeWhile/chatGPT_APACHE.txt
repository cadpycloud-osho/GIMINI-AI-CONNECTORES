import org.apache.camel.Exchange;
import org.apache.camel.Processor;

import java.util.LinkedHashMap;
import java.util.Map;

public class ObjectTakeWhileProcessor implements Processor {

    @Override
    public void process(Exchange exchange) {

        // IMPORTANT: LinkedHashMap preserves insertion order
        Map<String, Integer> obj = new LinkedHashMap<>();
        obj.put("a", 1);
        obj.put("b", 2);
        obj.put("c", 5);
        obj.put("d", 1);

        Map<String, Integer> result = new LinkedHashMap<>();

        for (Map.Entry<String, Integer> entry : obj.entrySet()) {
            if (entry.getValue() < 3) {
                result.put(entry.getKey(), entry.getValue());
            } else {
                // takeWhile â†’ STOP at first failure
                break;
            }
        }

        exchange.getIn().setBody(result);
    }
}

from("direct:start")
    .process(new ObjectTakeWhileProcessor())
    .to("log:result");
