import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Base64;

@SpringBootApplication
public class FromBase64DemoApplication {
    public static void main(String[] args) {
        SpringApplication.run(FromBase64DemoApplication.class, args);
    }
}

@RestController
class FromBase64Controller {
    
    @GetMapping(value = "/fromBase64", produces = MediaType.APPLICATION_OCTET_STREAM_VALUE)
    public byte[] fromBase64Example() {
        // Sample base64 payload: "SGVsbG8=" decodes to "Hello"
        String base64Payload = "SGVsbG8=";
        return Base64.getDecoder().decode(base64Payload);
    }
}
