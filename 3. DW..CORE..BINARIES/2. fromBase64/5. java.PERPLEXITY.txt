import java.util.Base64;
import java.nio.ByteBuffer;

public class Main {
    public static void main(String[] args) {
        // Input: Base64 string from previous example "yv6rzQ=="
        String base64Input = "yv6rzQ==";
        
        // DataWeave equivalent: fromBase64(payload)
        byte[] resultBytes = fromBase64(base64Input);
        
        // Output the result as hex (matches DataWeave binary output)
        System.out.println("Input Base64: " + base64Input);
        System.out.println("Output bytes (hex): " + bytesToHex(resultBytes));
    }
    
    // DataWeave "fromBase64(payload)" equivalent
    public static byte[] fromBase64(String base64String) {
        return Base64.getDecoder().decode(base64String);
    }
    
    // Utility to display bytes as hex
    public static String bytesToHex(byte[] bytes) {
        StringBuilder sb = new StringBuilder();
        for (byte b : bytes) {
            sb.append(String.format("%02X", b));
        }
        return sb.toString();
    }
}
