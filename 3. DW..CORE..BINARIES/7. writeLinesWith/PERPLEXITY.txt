import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.stream.IntStream;
import java.util.List;
import java.util.stream.Collectors;
import java.nio.charset.StandardCharsets;

@SpringBootApplication
public class WriteLinesDemoApplication {
    public static void main(String[] args) {
        SpringApplication.run(WriteLinesDemoApplication.class, args);
    }
}

@RestController
class WriteLinesController {
    
    @GetMapping("/writeLines")
    public Map<String, String> writeLinesExample() {
        // Generate lines: "Line 1" to "Line 10" (DataWeave to(1,10))
        List<String> lines = IntStream.rangeClosed(1, 10)
            .mapToObj(i -> "Line " + i)
            .collect(Collectors.toList());
        
        // Join with \n (UTF-8)
        String result = String.join("\n", lines) + "\n";
        
        Map<String, String> response = new HashMap<>();
        response.put("lines", result);
        
        return response;
    }
}
