import java.nio.charset.StandardCharsets;

public class Main {

    public static void main(String[] args) {

        // Equivalent of: "Mule" as Binary
        byte[] myBinary = "Mule".getBytes(StandardCharsets.UTF_8);

        // Convert binary to hex
        String binaryToHex = toHex(myBinary);

        // Output
        System.out.println("{ \"binaryToHex\": \"" + binaryToHex + "\" }");
    }

    // Helper method: byte[] â†’ hex string
    private static String toHex(byte[] bytes) {
        StringBuilder hex = new StringBuilder();
        for (byte b : bytes) {
            hex.append(String.format("%02X", b));
        }
        return hex.toString();
    }
}
