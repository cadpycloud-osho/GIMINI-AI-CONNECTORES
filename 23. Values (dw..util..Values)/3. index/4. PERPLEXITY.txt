@RestController
public class IndexController {
    
    @PostMapping("/create-index")
    public Map<String, Object> createIndex(@RequestBody Map<String, Integer> request) {
        int index = request.getOrDefault("index", 0);
        return createIndexObject(index);
    }
    
    private Map<String, Object> createIndexObject(int index) {
        Map<String, Object> indexSelector = new HashMap<>();
        indexSelector.put("kind", "Array");
        indexSelector.put("namespace", null);
        indexSelector.put("selector", index);
        return indexSelector;
    }
}
