public class IndexRepresentation {
    private String kind;
    private Object namespace; // Using Object to explicitly hold null
    private int selector;

    // Constructor to mimic the DataWeave function call
    public IndexRepresentation(int indexValue) {
        // The kind is fixed to "Array" for the 'index' function output
        this.kind = "Array"; 
        this.namespace = null;
        this.selector = indexValue;
    }

    // Getters (needed for Jackson serialization)
    public String getKind() {
        return kind;
    }
    public Object getNamespace() {
        return namespace;
    }
    public int getSelector() {
        return selector;
    }
}
