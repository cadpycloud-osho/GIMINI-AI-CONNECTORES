import java.util.LinkedHashMap;

public class SelectorMetadataRoute extends RouteBuilder {

    @Override
    public void configure() throws Exception {

        from("direct:getSelectorInfo")
            .routeId("Selector-Metadata-Route")
            .process(exchange -> {
                // 1. Create a structure mimicking the DW Selector Object
                Map<String, Object> selectorInfo = new LinkedHashMap<>();

                selectorInfo.put("namespace", null);
                selectorInfo.put("selector", 0);
