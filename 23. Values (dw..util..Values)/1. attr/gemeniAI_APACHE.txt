import java.util.LinkedHashMap;

public class DataWeaveAttrProcessor {
    public static Map<String, Object> attr(String namespaceUri, String attrName) {
        // DataWeave: attr(ns0, "myAttr")

            put("kind", "Attribute");
            put("namespace", namespaceUri);    // "http://acme.com/fo"
            put("selector", attrName);         // "myAttr"
        }};
    }
