import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class Main {

    // Target index is 1 (the second element in a 0-indexed list)
    private static final int TARGET_INDEX = 1; 
    // The value to replace the boolean with
    private static final Boolean MASK_VALUE = false;

    /**
     * Iterates over a List of Lists (inner arrays) and replaces the element 
     * at TARGET_INDEX with MASK_VALUE in each inner list.
     */
    public static List<List<Object>> maskInnerArrays(List<List<Object>> sourceData) {
        
        return sourceData.stream()
            .map(innerList -> {
                // 1. Create a mutable list. Arrays.asList() returns a fixed-size list 
                //    that cannot be modified, so we copy it to an ArrayList.
                List<Object> maskedList = new ArrayList<>(innerList);
                
                // 2. Check if the inner list is long enough to have the TARGET_INDEX
                if (maskedList.size() > TARGET_INDEX) {
                    // 3. Masking operation: replace the value at the index (index 1)
                    maskedList.set(TARGET_INDEX, MASK_VALUE);
                }
                return maskedList;
            })
            .collect(Collectors.toList());
    }

    public static void main(String[] args) {
        // 1. Source Data
        List<List<Object>> sourceData = Arrays.asList(
            Arrays.asList(123, true),
            Arrays.asList(456, true)
        );
        
        // 2. Apply transformation
        List<List<Object>> output = maskInnerArrays(sourceData);

        // 3. Print the output (Jackson dependencies removed, using standard Java printing)
        
        // Simple printing to match the requested output format: [ [ 123, false ], [ 456, false ] ]
        String simpleOutput = output.stream()
            .map(inner -> inner.toString().replace('[', '[').replace(']', ']').replace(", ", ", "))
            .collect(Collectors.joining(", ", "[ ", " ]"));

        System.out.println("Output (Pure Java List):");
        System.out.println(simpleOutput);
    }
}


output:

[ [ 123, false ], [ 456, false ] ]
