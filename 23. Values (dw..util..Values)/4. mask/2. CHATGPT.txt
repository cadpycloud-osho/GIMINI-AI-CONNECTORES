@RestController
@RequestMapping("/users")
public class UserController {

    @GetMapping("/masked")
    public List<Map<String, String>> getMaskedUsers() {

        List<Map<String, String>> users = new ArrayList<>();

        Map<String, String> u1 = new HashMap<>();
        u1.put("name", "Peter Parker");
        u1.put("password", "spiderman");

        Map<String, String> u2 = new HashMap<>();
        u2.put("name", "Bruce Wayne");
        u2.put("password", "batman");

        users.add(u1);
        users.add(u2);

        // Mask password
        users.forEach(user -> user.put("password", "*****"));

        return users;
    }
}


output:

[
  {
    "name": "Peter Parker",
    "password": "*****"
  },
  {
    "name": "Bruce Wayne",
    "password": "*****"
  }

package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.*;

@RestController
public class MaskController {

    @GetMapping("/mask")
    public List<List<Object>> maskValue() {

        List<List<Object>> data = new ArrayList<>();
        data.add(new ArrayList<>(Arrays.asList(123, true)));
        data.add(new ArrayList<>(Arrays.asList(456, true)));

        // Equivalent of: mask 1 with false
        for (List<Object> innerList : data) {
            if (innerList.size() > 1) {
                innerList.set(1, false);
            }
        }

        return data;
    }
}



output:


[
  [123, false],
  [456, false]
]
