    /**
     * Helper method to manually format the Map into a pretty-printed JSON string.
     */
    private static String formatMapAsPrettyJson(Map<String, Double> map) {
        StringBuilder sb = new StringBuilder();
        sb.append("{\n");
        
        // Assuming only one entry: "price"
        Map.Entry<String, Double> entry = map.entrySet().iterator().next();
        
        // Use full double precision to match the output style
        sb.append("  \"").append(entry.getKey()).append("\": ").append(entry.getValue()); 
        
        sb.append("\n}");
        return sb.toString();
    }
