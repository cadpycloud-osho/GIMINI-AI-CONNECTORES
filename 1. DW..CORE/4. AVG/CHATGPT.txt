avg(values: Array<Number>): Number
Returns the average of numbers listed in an array.

An array that is empty or that contains a non-numeric value results in an error.

INPUT:
package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.*;

@RestController
public class AvgController {

    @GetMapping("/avg")
    public Map<String, Double> getAverages() {

        List<Integer> listA = Arrays.asList(1, 1000);
        List<Integer> listB = Arrays.asList(1, 2, 3);

        double avgA = listA.stream().mapToInt(Integer::intValue).average().orElse(0);
        double avgB = listB.stream().mapToInt(Integer::intValue).average().orElse(0);

        Map<String, Double> response = new HashMap<>();
        response.put("a", avgA);
        response.put("b", avgB);

        return response;
    }
}


OUTPUT:
{
  "a": 500.5,
  "b": 2.0
}
