import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

@SpringBootApplication
public class FloorNumbersApp {

    public static void main(String[] args) {
        SpringApplication.run(FloorNumbersApp.class, args);

        List<Double> inputNumbers = Arrays.asList(1.5, 2.2, 3.0);

        List<Integer> flooredNumbers = inputNumbers.stream()
                .map(Math::floor)
                .map(Double::intValue)
                .collect(Collectors.toList());

        System.out.println(flooredNumbers);  // Output: [1, 2, 3]
    }
}
