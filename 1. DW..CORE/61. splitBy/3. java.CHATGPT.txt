splitBy:


example 1:

. "192.88.99.0/24" splitBy(/[.\/]/)
input:

import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        String input = "192.88.99.0/24";
        String[] result = input.split("[./]");
        System.out.println(Arrays.toString(result));
    }
}


output:

[192, 88, 99, 0, 24]

example 2:


input:

import java.util.Arrays;

public class Main {
    public static void main(String[] args) {

        String split1 = "a-b-c";
        String split2 = "hello world";
        String split3 = "no match";
        String split4 = "no match";
        String split5 = "a1b2c3d4A1B2C3D";

        System.out.println("split1: " + Arrays.toString(split1.split(".b.")));
        System.out.println("split2: " + Arrays.toString(split2.split("\\s")));
        System.out.println("split3: " + Arrays.toString(split3.split("^s")));
        System.out.println("split4: " + Arrays.toString(split4.split("^n..")));
        System.out.println("split5: " + Arrays.toString(split5.split("[0-9A-Z]")));
    }
}


output:

split1: [a, c]
split2: [hello, world]
split3: [no match]
split4: [, match]
split5: [a, b, c, d]

example 4:


input:

import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        String input = "root.sources.data.`test.branch.BranchSource`.source.traits";

        String regex = "[.](?=(?:[^`]*`[^`]*`)*[^`]*$)";
        String[] result = input.split(regex);

        System.out.println(Arrays.toString(result));
    }
}


output:

[root, sources, data, `test.branch.BranchSource`, source, traits]
