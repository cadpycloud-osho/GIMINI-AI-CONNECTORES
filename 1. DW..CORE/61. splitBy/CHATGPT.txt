example 2:


input:

@GetMapping("/split2")
public Map<String, Object> splitExample2() {

    Map<String, Object> map = new HashMap<>();

    map.put("split1", Arrays.asList("a-b-c".split(".b.")));
    map.put("split2", Arrays.asList("hello world".split("\\s")));
    map.put("split3", Arrays.asList("no match".split("^s"))); 
    map.put("split4", Arrays.asList("no match".split("^n.."))); 
    map.put("split5", Arrays.asList("a1b2c3d4A1B2C3D".split("[0-9A-Z]")));

    return Map.of("splitters", map);
}



output:
{
  "splitters": {
    "split1": ["a", "c"],
    "split2": ["hello", "world"],
    "split3": ["no match"],
    "split4": ["", "match"],
    "split5": ["a", "b", "c", "d"]
  }
}

example 6:


input:

@GetMapping("/split6")
public Map<String, Object> splitExample6() {

    Map<String, Object> map = new HashMap<>();
    map.put("split1", Arrays.asList("a-b-c".split("-")));
    map.put("split2", Arrays.asList("hello world".split("")));
    map.put("split3", Arrays.asList("first,middle,last".split(",")));
    map.put("split4", Arrays.asList("no split".split("NO")));

    return Map.of("splitters", map);
}



output:

{
  "splitters": {
    "split1": ["a", "b", "c"],
    "split2": ["h","e","l","l","o","","w","o","r","l","d"],
    "split3": ["first","middle","last"],
    "split4": ["no split"]
  }
}
