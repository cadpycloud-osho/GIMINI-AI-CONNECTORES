import java.util.Arrays;
import java.util.List;

public class Main {

    public static void main(String[] args) {

        List<Integer> listA = Arrays.asList(1, 1000);
        List<Integer> listB = Arrays.asList(1, 2, 3);
        List<Double> listC = Arrays.asList(1.5, 2.5, 3.5);

        int a = listA.stream().min(Integer::compareTo).orElseThrow();
        int b = listB.stream().min(Integer::compareTo).orElseThrow();
        double c = listC.stream().min(Double::compareTo).orElseThrow();

        System.out.println("{ \"a\": " + a + ", \"b\": " + b + ", \"c\": " + c + " }");
    }
}


output:



{ "a": 1, "b": 1, "c": 1.5 }
