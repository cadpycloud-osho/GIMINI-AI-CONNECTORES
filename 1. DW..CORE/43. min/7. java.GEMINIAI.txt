import java.util.Arrays;
import java.util.Collections;
import java.util.List;
import java.util.Locale;

public class Main {

    /**
     * Helper method to find the minimum value in a list of Integers.
     * DataWeave: min(list)
     */
    private static Integer minInt(List<Integer> list) {
        if (list == null || list.isEmpty()) {
            return null; // DataWeave min on empty array returns null
        }
        return Collections.min(list);
    }

    /**
     * Helper method to find the minimum value in a list of Doubles.
     * DataWeave: min(list)
     */
    private static Double minDouble(List<Double> list) {
        if (list == null || list.isEmpty()) {
            return null;
        }
        return Collections.min(list);
    }

    /**
     * DataWeave equivalent logic:
     * {
     * a: min([1, 1000]),
     * b: min([1, 2, 3]),
     * c: min([1.5, 2.5, 3.5])
     * }
     * Output: {"a":1,"b":1,"c":1.5}
     */
    public static void main(String[] args) throws Exception {
        // --- 1. Define input arrays ---
        List<Integer> arrayA = Arrays.asList(1, 1000);
        List<Integer> arrayB = Arrays.asList(1, 2, 3);
        List<Double> arrayC = Arrays.asList(1.5, 2.5, 3.5);

        // --- 2. Apply the 'min' function ---
        Integer resultA = minInt(arrayA); // Result: 1
        Integer resultB = minInt(arrayB); // Result: 1
        Double resultC = minDouble(arrayC); // Result: 1.5
        
        // Use Locale.US to ensure dot (.) as decimal separator for JSON output
        Locale.setDefault(Locale.US);

        // --- 3. Manually construct the final compact JSON object ---
        
        // Format for Integer key/value pairs
        String entryA = String.format("\"a\":%d", resultA);
        String entryB = String.format("\"b\":%d", resultB);

        // Format for Double key/value pairs
        String entryC = String.format("\"c\":%.1f", resultC);

        String jsonOutput = String.format("{%s,%s,%s}", entryA, entryB, entryC);
        
        System.out.println(jsonOutput);
    }
}
