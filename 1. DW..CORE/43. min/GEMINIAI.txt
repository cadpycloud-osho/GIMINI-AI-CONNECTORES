import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.Collections;
import java.util.LinkedHashMap;

@RestController
public class MinController {

    /**
     * Equivalent to: { a: min([1, 1000]), b: min([1, 2, 3]), c: min([1.5, 2.5, 3.5]) }
     * Accessible at http://localhost:8080/find-min
     */
    @GetMapping("/find-min")
    public Map<String, Number> findMinimums() {
        // Data setup: Lists of Integers and Doubles
        List<Integer> listA = Arrays.asList(1, 1000);
        List<Integer> listB = Arrays.asList(1, 2, 3);
        List<Double> listC = Arrays.asList(1.5, 2.5, 3.5);

        // 1. Calculate min for 'a'
        // Java equivalent of DataWeave: min([1, 1000]) -> 1
        Number minA = Collections.min(listA);

        // 2. Calculate min for 'b'
        // Java equivalent of DataWeave: min([1, 2, 3]) -> 1
        Number minB = Collections.min(listB);

        // 3. Calculate min for 'c'
        // Java equivalent of DataWeave: min([1.5, 2.5, 3.5]) -> 1.5
        Number minC = Collections.min(listC);

        // 4. Construct the JSON output map
        Map<String, Number> result = new LinkedHashMap<>();
        result.put("a", minA);
        result.put("b", minB);
        result.put("c", minC);

        return result;
    }
}
