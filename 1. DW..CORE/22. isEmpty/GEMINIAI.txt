import java.util.Arrays;
import java.util.Collections;
import java.util.List;

public class IsEmptyListExample {

    public static List<Boolean> checkListEmptiness() {
        // DataWeave: isEmpty([])
        List<Integer> emptyList = Collections.emptyList();
        boolean result1 = emptyList.isEmpty(); 

        // DataWeave: isEmpty([1])
        List<Integer> populatedList = Arrays.asList(1);
        boolean result2 = populatedList.isEmpty();

        // Return results as a List (JSON array equivalent)
        return Arrays.asList(result1, result2);
    }

    public static void main(String[] args) {
        List<Boolean> output = checkListEmptiness();
        System.out.println("Input Logic: List.isEmpty()");
        System.out.println("Output (JSON equivalent): " + output);
    }
}
OUTPUT:


[true, false]

import java.util.Map;

public class IsEmptyNullExample {

    // Custom check to replicate DW's isEmpty(null) -> true
    public static boolean isNullEmpty(Object value) {
        return value == null;
    }

    public static Map<String, Boolean> checkNullEmptiness() {
        // DataWeave: isEmpty(null)
        Object nullInput = null;
        boolean result = isNullEmpty(nullInput);
        
        // Return results as a Map (JSON object equivalent)
        return Map.of("nullValue", result);
    }

    public static void main(String[] args) {
        Map<String, Boolean> output = checkNullEmptiness();
        System.out.println("Input Logic: value == null");
        System.out.println("Output (JSON equivalent): " + output);
    }
}

OUTPUT:


{nullValue=true}
