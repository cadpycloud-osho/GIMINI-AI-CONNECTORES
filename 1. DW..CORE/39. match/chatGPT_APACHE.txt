from("direct:flatMapMatch")
.setBody(constant(List.of(
    "192.88.99.0/24",
    "192.168.0.0/16",
    "192.175.48.0/24"
)))

.split(body())
    .filter(simple("${body} regex '.*\\/24$'"))

        .completionTimeout(100)
.end();

[
  "192.88.99.0/24",
  "192.175.48.0/24"
