package com.example.demo;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import org.json.JSONObject;
import org.json.XML;

@SpringBootApplication
@RestController
public class JsonToXmlController {

    public static void main(String[] args) {
        SpringApplication.run(JsonToXmlController.class, args);
    }

    @GetMapping(value = "/json-to-xml", produces = MediaType.APPLICATION_XML_VALUE)
    public String convertJsonToXml() {
        String json = "{ \"hello\" : \"world\" }";
        JSONObject jsonObj = new JSONObject(json);
        return XML.toString(jsonObj, "hello");
    }
}


output:

<?xml version='1.0' encoding='UTF-8'?>
<hello>world</hello>

package com.example.demo;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class CsvToJsonController {

    @GetMapping("/csv-to-json")
    public Map<String, String> convertCsvToJson() {
        String input = "Some, Body";
        String[] parts = input.split(",");

        Map<String, String> result = new LinkedHashMap<>();
        result.put("column_0", parts[0]);
        result.put("column_1", parts[1]);

        return result;
    }
}


output:

{
  "column_0": "Some",
  "column_1": " Body"
}
