isLeapYear:

EXAMPLE 1:

INPUT:

import java.time.LocalDateTime;
import java.util.Arrays;
import java.util.List;

public class IsLeapYearExample1 {

    public static List<Boolean> checkLocalDateTime() {
        // DataWeave: |2016-10-01T23:57:59| -> LocalDateTime
        LocalDateTime dt2016 = LocalDateTime.parse("2016-10-01T23:57:59");
        // DataWeave: |2017-10-01T23:57:59| -> LocalDateTime
        LocalDateTime dt2017 = LocalDateTime.parse("2017-10-01T23:57:59");
        
        // Get the year and check if it's a leap year
        boolean is2016Leap = dt2016.toLocalDate().isLeapYear(); // true
        boolean is2017Leap = dt2017.toLocalDate().isLeapYear(); // false

        return Arrays.asList(is2016Leap, is2017Leap);
    }

    public static void main(String[] args) {
        List<Boolean> output = checkLocalDateTime();
        System.out.println("Input Logic: isLeapYear on LocalDateTime");
        System.out.println("Output (JSON equivalent): " + output);
    }
}

OUTPUT:

[true, false]

EXAMPLE 3:

INPUT:

import java.time.OffsetDateTime;
import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class IsLeapYearExample3 {

    public static List<Boolean> checkOffsetDateTime() {
        // DataWeave: |2016-10-01T23:57:59-03:00| -> OffsetDateTime
        OffsetDateTime dtOffset = OffsetDateTime.parse("2016-10-01T23:57:59-03:00");
        // DataWeave: |2016-10-01T23:57:59Z| -> OffsetDateTime (Z is shortcut for +00:00)
        OffsetDateTime dtZ = OffsetDateTime.parse("2016-10-01T23:57:59Z");
        
        List<OffsetDateTime> dates = Arrays.asList(dtOffset, dtZ);
        
        // Map the operation: get the LocalDate from the DateTime and check its year
        List<Boolean> results = dates.stream()
            .map(dateTime -> dateTime.toLocalDate().isLeapYear())
            .collect(Collectors.toList());

        return results;
    }

    public static void main(String[] args) {
        List<Boolean> output = checkOffsetDateTime();
        System.out.println("Input Logic: isLeapYear on OffsetDateTime/ZonedDateTime");
        System.out.println("Output (JSON equivalent): " + output);
    }
}
OUTPUT:

[true, true]
