package com.example.demo;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.time.ZonedDateTime;
import java.util.*;

@RestController
public class LeapYear1Controller {

    @GetMapping("/leap1")
    public List<Boolean> checkLeapYears() {
        ZonedDateTime dt1 = ZonedDateTime.parse("2016-10-01T23:57:59Z");
        ZonedDateTime dt2 = ZonedDateTime.parse("2017-10-01T23:57:59Z");

        return List.of(dt1.toLocalDate().isLeapYear(), dt2.toLocalDate().isLeapYear());
    }
}
