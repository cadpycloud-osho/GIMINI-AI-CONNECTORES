import java.util.List;
import java.util.stream.Collectors;

public class Main {
    
    /**
     * Helper method to manually format the List of Booleans into a single-line JSON array string.
     */
    private static String formatListAsJsonArray(List<Boolean> list) {
        // Collect booleans and join with ", "
        String elements = list.stream()
            .map(String::valueOf)
            .collect(Collectors.joining(", "));
            
        // The requested output is formatted as [ true, false ]
        return "[ " + elements + " ]";
    }
