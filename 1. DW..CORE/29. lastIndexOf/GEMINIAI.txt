import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.LinkedHashMap;

@RestController
public class LastIndexOfController {

    /**
     * Equivalent to DataWeave lastIndexOf on Arrays/Lists.
     * Accessible at http://localhost:8080/list-last-index
     */
    @GetMapping("/list-last-index")
    public Map<String, Integer> listLastIndexOf() {
        // Data setup
        List<String> list1 = Arrays.asList("a", "b", "c", "d");
        List<String> list2 = Arrays.asList("x", "w", "x");
        List<String> list3 = Arrays.asList("a", "b", "c", "c");

        // Use List.lastIndexOf(Object o)
        int present = list1.lastIndexOf("c");          // Output: 2 (c is at index 2)
        int notPresent = list2.lastIndexOf("c");       // Output: -1 (c is not present)
        int presentMoreThanOnce = list3.lastIndexOf("c"); // Output: 3 (the *last* c is at index 3)

        // Construct the JSON output
        Map<String, Integer> result = new LinkedHashMap<>();
        result.put("present", present);
        result.put("notPresent", notPresent);
        result.put("presentMoreThanOnce", presentMoreThanOnce);

        return result;
    }

    // ... String Last Index Method will go here ...
}


output:


{
  "present": 2,
  "notPresent": -1,
  "presentMoreThanOnce": 3
}
