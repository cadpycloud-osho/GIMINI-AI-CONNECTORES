import java.util.Arrays;
import java.util.List;

public class Main {

    // --- Utility Methods to replicate DataWeave's isInteger() behavior ---

    /**
     * Handles standard integer types (e.g., isInteger(1)).
     */
    public static boolean isInteger(int number) {
        return true;
    }

    /**
     * Handles floating-point types (e.g., isInteger(2.0) and isInteger(2.2)).
     * Returns true only if the floating-point number is mathematically an integer (i.e., has no fractional part).
     */
    public static boolean isInteger(double number) {
        // Check if the number is equal to its closest long (integer) representation.
        // This is the standard way to check for a zero fractional part.
        return number == Math.floor(number);
    }

    /**
     * Handles String types (e.g., isInteger("1")).
     * Returns true if the string can be parsed as an integer.
     */
    public static boolean isInteger(String str) {
        if (str == null || str.trim().isEmpty()) {
            return false;
        }
        try {
            // Attempt to parse the string as an Integer
            Integer.parseInt(str.trim());
            return true;
        } catch (NumberFormatException e) {
            // If the string contains anything non-integer (like "2.2" or "a"), parsing fails.
            return false;
        }
    }

    public static void main(String[] args) {

        // --- DataWeave Expressions to be evaluated ---

        // isInteger(1) -> Calls isInteger(int)
        boolean result1 = isInteger(1);

        // isInteger(2.0) -> Calls isInteger(double)
        boolean result2 = isInteger(2.0);

        // isInteger(2.2) -> Calls isInteger(double)
        boolean result3 = isInteger(2.2);

        // isInteger("1") -> Calls isInteger(String)
        boolean result4 = isInteger("1");

        // Collect the results into a List (equivalent to the JSON array output)
        List<Boolean> outputList = Arrays.asList(result1, result2, result3, result4);
