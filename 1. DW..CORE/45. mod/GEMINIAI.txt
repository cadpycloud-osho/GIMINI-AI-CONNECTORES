import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.List;
import java.util.Arrays;
import java.util.Collections;

@RestController
public class ModuloController {

    /**
     * Equivalent to: [ (3 mod 2), (4 mod 2), (2.2 mod 2) ]
     * Accessible at http://localhost:8080/modulo-examples
     */
    @GetMapping("/modulo-examples")
    public List<Number> calculateModulo() {
        
        // 1. Integer Modulo: (3 mod 2) -> 3 % 2
        int mod1 = 3 % 2; // Result: 1

        // 2. Integer Modulo: (4 mod 2) -> 4 % 2
        int mod2 = 4 % 2; // Result: 0

        // 3. Floating-Point Modulo: (2.2 mod 2) -> 2.2 % 2.0
        // Java's % operator works for floating-point numbers.
        double mod3 = 2.2 % 2.0; // Result: 0.2
        
        // Return the results as a List<Number> (to accommodate both int and double)
        return Arrays.asList(mod1, mod2, mod3);
    }
}
