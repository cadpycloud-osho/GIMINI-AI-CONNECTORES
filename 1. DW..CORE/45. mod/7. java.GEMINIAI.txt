import java.util.List;
import java.util.stream.Collectors;

public class Main {

    /**
     * DataWeave equivalent logic:
     * [ 3 mod 2, 4 mod 2, 2.2 mod 2 ]
     * Output: [ 1, 0, 0.2 ]
     */
    public static void main(String[] args) throws Exception {
        // --- 1. Define the operations ---
        
        // Operation 1: 3 mod 2 (Integer operation)
        int result1 = 3 % 2; // Result: 1

        // Operation 2: 4 mod 2 (Integer operation)
        int result2 = 4 % 2; // Result: 0

        // Operation 3: 2.2 mod 2 (Floating-point operation)
        // Note: Java's % operator works on doubles to find the remainder.
        double result3 = 2.2 % 2; // Result: 0.2

        // --- 2. Combine the results into a List ---
        // Note: We use Object to hold the mix of Integer and Double results
        List<Object> results = List.of(result1, result2, result3);

        // --- 3. Manually construct the final JSON array string ---
        
        String joinedResults = results.stream()
            .map(value -> {
                // Check if the value is an Integer or a Double for formatting
                if (value instanceof Double) {
                    // Use String.format for precise double representation (e.g., to ensure 0.2 is not 0.200000001)
                    return String.format("%.1f", value); // 0.2
                } else {
                    return value.toString(); // 1 or 0
                }
            })
            .collect(Collectors.joining(", "));
        
        String outputJson = String.format("[ %s ]", joinedResults);
        
        System.out.println(outputJson);
    }
}
