package com.example.demo;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.*;

@SpringBootApplication
@RestController
public class ModOperationController {

    public static void main(String[] args) {
        SpringApplication.run(ModOperationController.class, args);
    }

    @GetMapping("/mod-operation")
    public List<Double> getModOperation() {

        List<Double> result = Arrays.asList(
            3 % 2.0,     // 3 mod 2 = 1
            4 % 2.0,     // 4 mod 2 = 0
            2.2 % 2.0    // 2.2 mod 2 = 0.2
        );

        return result;
    }
}
