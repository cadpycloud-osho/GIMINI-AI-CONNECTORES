import java.util.*;
import java.util.Map.Entry;

public class Main {

    public static void main(String[] args) {

        List<Map<String, Integer>> list = Arrays.asList(
                Collections.singletonMap("a", 1),
                Collections.singletonMap("a", 2),
                Collections.singletonMap("a", 3)
        );

        Map<String, Integer> min = list.stream()
                .min(Comparator.comparingInt(item -> item.get("a")))
                .orElse(null);

        System.out.println(min);
    }
}

example 2:


input:

import java.time.*;
import java.util.*;

public class Main {

    public static void main(String[] args) {

        String myDateTime1 = "2017-10-01T22:57:59-03:00";
        String myDateTime2 = "2018-10-01T23:57:59-03:00";

        Map<String, Object> myMinBy = new LinkedHashMap<>();

        // byDateTime (ISO-8601 strings compare correctly)
        List<String> dateTimeList = Arrays.asList(myDateTime1, myDateTime2);
        String minByDateTime =
                dateTimeList.stream().min(String::compareTo).orElse(null);
        myMinBy.put("byDateTime", minByDateTime);

        // byDate
        List<LocalDate> dateList = Arrays.asList(
                OffsetDateTime.parse(myDateTime1).toLocalDate(),
                OffsetDateTime.parse(myDateTime2).toLocalDate()
        );
        LocalDate minByDate =
                dateList.stream().min(LocalDate::compareTo).orElse(null);
        myMinBy.put("byDate", minByDate);

        // byTime (FIXED HERE)
        List<OffsetTime> timeList = Arrays.asList(
                OffsetDateTime.parse(myDateTime1).toOffsetTime(),
                OffsetDateTime.parse(myDateTime2).toOffsetTime()
        );
        OffsetTime minByTime =
                timeList.stream().min(OffsetTime::compareTo).orElse(null);
        myMinBy.put("byTime", minByTime);

        // aBoolean
        List<Boolean> boolList = Arrays.asList(true, false, (0 > 1), (1 > 0));
        Boolean minByBool =
                boolList.stream().min(Boolean::compareTo).orElse(null);
        myMinBy.put("aBoolean", minByBool);

        // emptyArray
        List<Integer> emptyList = new ArrayList<>();
        Integer minByEmpty =
                emptyList.stream().min(Integer::compareTo).orElse(null);
        myMinBy.put("emptyArray", minByEmpty);

        Map<String, Object> output = new LinkedHashMap<>();
        output.put("myMinBy", myMinBy);

        System.out.println(output);
    }
}


output:


{
  myMinBy={
    byDateTime=2017-10-01T22:57:59-03:00, 
    byDate=2017-10-01, 
    byTime=22:57:59-03:00, 
    aBoolean=false, 
    emptyArray=null
  }
}
