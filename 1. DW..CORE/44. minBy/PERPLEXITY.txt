// MinByObjectApp.java
package com.example.demo;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.util.*;
import java.util.Comparator;

@SpringBootApplication
public class MinByObjectApp {
    public static void main(String[] args) {
        SpringApplication.run(MinByObjectApp.class, args);

        List<Map<String, Integer>> items = Arrays.asList(
            Map.of("a", 1),
            Map.of("a", 2),
            Map.of("a", 3)
        );

        Map<String, Integer> minItem = Collections.min(items, Comparator.comparingInt(item -> item.get("a")));
        System.out.println(minItem);  // Output: {a=1}
    }
}


OUTPUT:

{a=1}

EXAMPLE 2:

INPUT:


// MinByDateTimeApp.java
package com.example.demo;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.time.*;
import java.time.format.DateTimeFormatter;
import java.util.*;

@SpringBootApplication
public class MinByDateTimeApp {
    public static void main(String[] args) {
        SpringApplication.run(MinByDateTimeApp.class, args);

        OffsetDateTime dt1 = OffsetDateTime.parse("2017-10-01T22:57:59-03:00");
        OffsetDateTime dt2 = OffsetDateTime.parse("2018-10-01T23:57:59-03:00");

        OffsetDateTime minDateTime = Collections.min(Arrays.asList(dt1, dt2));
        LocalDate minDate = Collections.min(Arrays.asList(dt1.toLocalDate(), dt2.toLocalDate()));
        OffsetTime minTime = Collections.min(Arrays.asList(dt1.toOffsetTime(), dt2.toOffsetTime()));
        Boolean minBoolean = Collections.min(Arrays.asList(true, false, false, true));
        Object emptyMin = null; // DataWeave returns null for empty array

        Map<String, Object> myMinBy = Map.of(
            "byDateTime", minDateTime.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME),
            "byDate", minDate.format(DateTimeFormatter.ISO_LOCAL_DATE),
            "byTime", minTime.format(DateTimeFormatter.ISO_OFFSET_TIME),
            "aBoolean", minBoolean,
            "emptyArray", emptyMin
        );

        Map<String, Object> result = Map.of("myMinBy", myMinBy);
        System.out.println(result);
    }
}


OUTPUT:

{myMinBy={byDateTime=2017-10-01T22:57:59-03:00, byDate=2017-10-01, byTime=22:57:59-03:00, aBoolean=false, emptyArray=null}}
