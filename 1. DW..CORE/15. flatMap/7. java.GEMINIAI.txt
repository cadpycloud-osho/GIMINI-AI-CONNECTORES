import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class Main {

    public static void main(String[] args) {
        // DataWeave Source: [ [3,5], [0.9,5.5] ] flatMap (value, index) -> value

        // 1. Define the input List of Lists (representing the nested array)
        // We use Double to accommodate both integer (3, 5) and decimal (0.9, 5.5) inputs.
        List<List<Double>> sourceList = Arrays.asList(
            Arrays.asList(3.0, 5.0),
            Arrays.asList(0.9, 5.5)
        );

        // 2. Core Logic: Use Stream.flatMap()
        // The DataWeave criteria (value, index) -> value is equivalent to
        // mapping each inner list to its own stream (List::stream) and flattening it.
        List<Double> flatList = sourceList.stream()
            // flatMap takes each inner List<Double> and transforms it into a Stream<Double>
            // before combining all streams into the result.
            .flatMap(List::stream)
            .collect(Collectors.toList());

        // Print the output
        System.out.println("Output:");
        System.out.println(flatList);
    }
}
