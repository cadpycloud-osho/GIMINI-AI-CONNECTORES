import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

@SpringBootApplication
public class FlatMapApp {

    public static void main(String[] args) {
        SpringApplication.run(FlatMapApp.class, args);

        List<List<Number>> nestedList = Arrays.asList(
                Arrays.asList(3, 5),
                Arrays.asList(0.9, 5.5)
        );

        List<Number> flattened = nestedList.stream()
                .flatMap(List::stream)
                .collect(Collectors.toList());

        System.out.println(flattened); // Output: [3, 5, 0.9, 5.5]
    }
}
