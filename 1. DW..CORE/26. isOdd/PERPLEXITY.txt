import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

@SpringBootApplication
public class IsOddApp {

    public static void main(String[] args) {
        SpringApplication.run(IsOddApp.class, args);

        List<Boolean> isOddResults = Arrays.asList(
                isOdd(0),           // false
                isOdd(1),           // true  
                isOdd(2 + 2)        // false (4)
        );

        Map<String, List<Boolean>> result = Map.of("isOdd", isOddResults);

        System.out.println(result); // Output: {isOdd=[false, true, false]}
    }

    private static boolean isOdd(int number) {
        return number % 2 != 0;
    }

    private static boolean isOdd(double number) {
        return (int) Math.floor(number) % 2 != 0;
    }
}
