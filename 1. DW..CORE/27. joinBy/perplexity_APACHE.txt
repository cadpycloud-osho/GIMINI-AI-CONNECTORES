import org.apache.camel.Exchange;
import org.apache.camel.Processor;
import java.util.List;
import com.fasterxml.jackson.annotation.JsonProperty;

public class JoinByProcessor implements Processor {
    @Override
    public void process(Exchange exchange) throws Exception {
        List<String> inputArray = exchange.getIn().getBody(List.class);
        String result = String.join("-", inputArray);
        
        exchange.getIn().setBody(new JoinByResult(result));
    }
    
    public static class JoinByResult {
        @JsonProperty("hyphenate")
        public String hyphenate;
        
        public JoinByResult(String hyphenate) {
            this.hyphenate = hyphenate;
        }
    }
}
