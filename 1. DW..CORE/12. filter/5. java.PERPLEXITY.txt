import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class Main {
    public static void main(String[] args) {
        List<Integer> numbers = Arrays.asList(9, 2, 3, 4, 5);
        List<Integer> result = numbers.stream()
            .filter(value -> value > 2)
            .collect(Collectors.toList());
        System.out.println(result);
    }
}



output:

[9, 3, 4, 5]

import java.util.*;
import java.util.stream.Collectors;

public class Main {
    public static void main(String[] args) {
        record Person(String name, int age) {}
        
        List<Person> people = Arrays.asList(
            new Person("Mariano", 37),
            new Person("Shoki", 30),
            new Person("Tomo", 25),
            new Person("Ana", 29)
        );
        
        List<Person> result = people.stream()
            .filter(person -> person.age() >= 30)
            .collect(Collectors.toList());
        System.out.println(result);
    }
}


output:

[{"name":"Mariano","age":37},{"name":"Shoki","age":30}]

import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.IntStream;

public class Main {
    public static void main(String[] args) {
        List<Integer> numbers = Arrays.asList(9, 2, 3, 4, 5);
        List<Integer> result = IntStream.range(0, numbers.size())
            .filter(index -> index > 1 && numbers.get(index) < 5)
            .mapToObj(numbers::get)
            .collect(Collectors.toList());
        System.out.println(result);
    }
}





output:

[3, 4]

import java.util.*;
import java.util.stream.Collectors;

public class Main {
    public static void main(String[] args) {
        record User(Map<String, String> user, String error) {}
        
        List<User> users = Arrays.asList(
            new User(Map.of("name", "123", "lastName", "Smith"), "That name doesn't exists"),
            new User(Map.of("name", "John", "lastName", "Johnson"), null)
        );
        
        List<User> result = users.stream()
            .filter(user -> user.error() == null)
            .collect(Collectors.toList());
        System.out.println(result);
    }
}


output:

[{"user":{"name":"John","lastName":"Johnson"},"error":null}]

import java.util.stream.Collectors;
import java.util.stream.IntStream;

public class Main {
    public static void main(String[] args) {
        String text = "hello world";
        String result = IntStream.range(0, text.length())
            .filter(index -> index % 2 == 0)
            .mapToObj(i -> String.valueOf(text.charAt(i)))
            .collect(Collectors.joining());
        System.out.println("\"" + result + "\"");
    }
}




output:


"hlowrd"
