input:

import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class Main {

    public static void main(String[] args) {

        List<String> list1 = Arrays.asList("a", "b", "c", "d");
        List<String> list2 = Arrays.asList("x", "w", "x");
        List<String> list3 = Arrays.asList("a", "b", "c", "c");

        Map<String, Integer> result = new HashMap<>();

        result.put("present", list1.indexOf("c"));
        result.put("notPresent", list2.indexOf("c"));
        result.put("presentMoreThanOnce", list3.indexOf("c"));

        System.out.println(result);
    }
}


output:

{present=2, notPresent=-1, presentMoreThanOnce=2}
