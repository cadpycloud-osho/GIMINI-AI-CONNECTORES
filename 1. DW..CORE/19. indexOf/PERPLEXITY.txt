import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.util.List;

@SpringBootApplication
public class ListIndexOfApp {

    public static void main(String[] args) {
        SpringApplication.run(ListIndexOfApp.class, args);

        List<String> presentList = List.of("a", "b", "c", "d");
        List<String> notPresentList = List.of("x", "w", "x");
        List<String> presentMoreThanOnceList = List.of("a", "b", "c", "c");

        int presentIndex = presentList.indexOf("c");
        int notPresentIndex = notPresentList.indexOf("c");
        int presentMoreThanOnceIndex = presentMoreThanOnceList.indexOf("c");

        System.out.println("{");
        System.out.println("  \"present\": " + presentIndex + ",");
        System.out.println("  \"notPresent\": " + notPresentIndex + ",");
        System.out.println("  \"presentMoreThanOnce\": " + presentMoreThanOnceIndex);
        System.out.println("}");
        // Output:
        // {
        //   "present": 2,
        //   "notPresent": -1,
        //   "presentMoreThanOnce": 2
        // }
    }
}
