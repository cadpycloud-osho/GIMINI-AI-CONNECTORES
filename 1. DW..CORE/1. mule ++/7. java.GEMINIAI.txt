input:

public class Example1 {
    public static void main(String[] args) {
        List<Object> list1 = new ArrayList<>(List.of(0, 1, true, "my string"));
        list1.addAll(List.of(2, List.of(3,4,5), Map.of("a",6)));

        System.out.println(Map.of("a", list1));
    }
}


output:

{
  "a": [0, 1, true, "my string", 2, [3, 4, 5], { "a": 6 }]
}

public class Example3 {
    public static void main(String[] args) {
        ZonedDateTime zdt =
            ZonedDateTime.of(
                LocalDate.of(2017,10,1),
                LocalTime.of(23,57,59),
                ZoneOffset.UTC);

        System.out.println(zdt);
    }
}

{
  "DateTime1": "2017-10-01T23:57:59Z"
}
