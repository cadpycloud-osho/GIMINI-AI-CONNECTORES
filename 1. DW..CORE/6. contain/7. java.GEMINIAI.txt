import java.util.Map;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class Main { // Renamed the class to 'Main' to resolve the compilation error

    public static void main(String[] args) {
        // 1. Simulate reading the content of the mystring XML tag
        String sourceString = "A very long string"; // Value of payload.root.mystring

        // 2. Define the Regular Expression Pattern
        // The DW pattern /s[t|p]rin/ means: "s" followed by "t" OR "p", followed by "rin".
        // The pattern 's[t|p]rin' matches 'strin' in 'string'.
        String regexPattern = "s[t|p]rin"; 
        
        // 3. Core Logic: Use java.util.regex
        Pattern pattern = Pattern.compile(regexPattern);
        Matcher matcher = pattern.matcher(sourceString);

        // The find() method checks if any part of the sourceString matches the pattern.
        boolean containsPattern = matcher.find(); // True, because "strin" is found.

        // 4. Construct the output Map
        // Uses Java 9+ Map.of()
        Map<String, Boolean> output = Map.of("ContainsString", containsPattern);

        // Print the output
        System.out.println("Output:");
        System.out.println(output);
    }
}

output:

{ContainsString: true}
