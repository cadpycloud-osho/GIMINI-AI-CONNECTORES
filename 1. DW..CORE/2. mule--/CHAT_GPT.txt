package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.*;

@RestController
public class DataWeaveController {

    @GetMapping("/subtractList")
    public Map<String, List<Integer>> subtractList() {

        List<Integer> listA = new ArrayList<>(Arrays.asList(0, 1, 1, 2));
        List<Integer> listB = Arrays.asList(1, 2);

        listA.removeAll(listB); // Same as DataWeave: [0,1,1,2] -- [1,2]

        Map<String, List<Integer>> response = new HashMap<>();
        response.put("a", listA);
        
        return response;
    }
}

OUTPUT:
{
  "good": "bad"
}

Removes specified key-value pairs from an object.
INPUT:

package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

@RestController
public class DataWeaveKeyController {

    @GetMapping("/removeKey")
    public Map<String, String> removeKey() {

        // Original Map like DataWeave object
        Map<String, String> map = new HashMap<>();
        map.put("hello", "world");
        map.put("name", "DW");

        // Remove the key "hello"
        map.remove("hello");

        return map; // Returns { "name": "DW" }
    }
}

OUTPUT:
{
  "name": "DW"
}
