import org.springframework.boot.SpringApplication;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Map;
import java.util.List;
import java.util.stream.Collectors;

@SpringBootApplication
public class DemoApplication {
    public static void main(String[] args) {
        SpringApplication.run(DemoApplication.class, args);
    }
}

@RestController
class DataWeaveController {
    @GetMapping("/dataweave-object-difference")
    public Map<String, String> getObjectDifference() {
        Map<String, String> original = Map.of(
            "yes", "no", 
            "good", "bad", 
            "old", "new"
        );
        
        List<String> keysToRemove = List.of("yes", "old");
        
        Map<String, String> result = original.entrySet().stream()
            .filter(entry -> !keysToRemove.contains(entry.getKey()))
            .collect(Collectors.toMap(
                Map.Entry::getKey, 
                Map.Entry::getValue
            ));
            
        return result;
    }
}


OUTPUT:


{
  "good": "bad"
}
