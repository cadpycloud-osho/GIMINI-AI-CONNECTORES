Alternative using BigDecimal for precise decimal handling:

java
import java.math.BigDecimal;
import java.math.RoundingMode;

public class RoundBigDecimal {
    public static void main(String[] args) {
        int round1 = BigDecimal.valueOf(1.2).setScale(0, RoundingMode.HALF_EVEN).intValue();
        int round2 = BigDecimal.valueOf(4.6).setScale(0, RoundingMode.HALF_EVEN).intValue();
        int round3 = BigDecimal.valueOf(3.5).setScale(0, RoundingMode.HALF_EVEN).intValue();
        
        System.out.println("[" + round1 + ", " + round2 + ", " + round3 + "]");
    }
}
