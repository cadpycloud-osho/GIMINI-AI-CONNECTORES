import org.apache.camel.CamelContext;
import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;
import org.apache.camel.model.dataformat.JsonLibrary;
import java.util.Arrays;
import java.util.List;

public class CamelRegexReplace {
    public static void main(String[] args) throws Exception {
        CamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {
                from("direct:start")
                    .process(exchange -> {
                        // 1. Strip everything before the last hyphen
                        // DataWeave: replace /.*-/ with("")
                        String val1 = "123-456-7890".replaceAll(".*-", "");
                        
                        // 2. Replace specific characters in a class
                        // DataWeave: replace /[b13e]/ with("-")
                        String val2 = "abc123def".replaceAll("[b13e]", "-");

                        List<String> result = Arrays.asList(val1, val2);
                        exchange.getIn().setBody(result);
                    })
                    .marshal().json(JsonLibrary.Jackson)
                    .to("stream:out");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(1000);
        context.stop();
    }
}

output:

[ "7890", "a-c-2-d-f" ]

example 2:


input:

import org.apache.camel.CamelContext;
import org.apache.camel.builder.RouteBuilder;
import org.apache.camel.impl.DefaultCamelContext;
import org.apache.camel.model.dataformat.JsonLibrary;
import java.util.Map;

public class CamelSimpleReplace {
    public static void main(String[] args) throws Exception {
        CamelContext context = new DefaultCamelContext();

        context.addRoutes(new RouteBuilder() {
            @Override
            public void configure() {
                from("direct:start")
                    .process(exchange -> {
                        String original = "admin123";
                        
                        // DataWeave: "admin123" replace "123" with("ID")
                        String replaced = original.replace("123", "ID");

                        // Wrap in a Map for JSON output
                        exchange.getIn().setBody(Map.of("replace", replaced));
                    })
                    .marshal().json(JsonLibrary.Jackson)
                    .to("stream:out");
            }
        });

        context.start();
        context.createProducerTemplate().sendBody("direct:start", null);
        Thread.sleep(1000);
        context.stop();
    }
}

output:

{ "replace": "adminID" }
