replace:

example 1:


input:

from("direct:start")
    .setBody(constant(java.util.Arrays.asList("123-456-7890", "abc123def")))
    .process(exchange -> {
        java.util.List<String> list = exchange.getIn().getBody(java.util.List.class);
        java.util.List<Object> result = new java.util.ArrayList<>();

        // First replacement: remove everything up to last dash
        result.add(list.get(0).replaceAll(".*-", ""));

        // Second replacement: replace b,1,3,e with "-"
        result.add(list.get(1).replaceAll("[b13e]", "-"));

        exchange.getIn().setBody(result);
    })
    .log("Output: ${body}");


output:


[ "7890", "a-c-2-d-f" ]

example 2:


input:

from("direct:start")
    .setBody(constant("admin123"))
    .process(exchange -> {
        String input = exchange.getIn().getBody(String.class);
        java.util.Map<String, String> result = new java.util.HashMap<>();
        result.put("replace", input.replace("123", "ID"));
        exchange.getIn().setBody(result);
    })
    .log("Output: ${body}");


output:

{ "replace": "adminID" }
