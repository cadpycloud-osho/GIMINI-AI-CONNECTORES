import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.Collections;
import java.util.LinkedHashMap;

@RestController
public class MaxController {

    /**
     * Equivalent to: { a: max([1, 1000]), b: max([1, 2, 3]), c: max([1.5, 2.5, 3.5]) }
     * Accessible at http://localhost:8080/find-max
     */
    @GetMapping("/find-max")
    public Map<String, Number> findMaximums() {
        // Data setup: Lists of Integers and Doubles
        List<Integer> listA = Arrays.asList(1, 1000);
        List<Integer> listB = Arrays.asList(1, 2, 3);
        List<Double> listC = Arrays.asList(1.5, 2.5, 3.5);

        // 1. Calculate max for 'a'
        // DataWeave: max([1, 1000]) -> 1000
        Number maxA = Collections.max(listA);

        // 2. Calculate max for 'b'
        // DataWeave: max([1, 2, 3]) -> 3
        Number maxB = Collections.max(listB);

        // 3. Calculate max for 'c'
        // DataWeave: max([1.5, 2.5, 3.5]) -> 3.5
        Number maxC = Collections.max(listC);

        // 4. Construct the JSON output map
        Map<String, Number> result = new LinkedHashMap<>();
        result.put("a", maxA);
        result.put("b", maxB);
        result.put("c", maxC);

        return result;
    }
}
