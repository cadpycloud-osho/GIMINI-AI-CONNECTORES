example 1:


input:

import java.time.ZonedDateTime;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;

public class Main {
    public static void main(String[] args) {
        // Current time in system default timezone
        ZonedDateTime nowCalled = ZonedDateTime.now();

        // Current time in specific timezone (America/New_York)
        ZonedDateTime nowCalledSpecificTimeZone = ZonedDateTime.now(ZoneId.of("America/New_York"));

        // Print JSON-like output
        System.out.println("{");
        System.out.println("  \"nowCalled\": \"" + nowCalled + "\",");
        System.out.println("  \"nowCalledSpecificTimeZone\": \"" + nowCalledSpecificTimeZone + "\"");
        System.out.println("}");
    }
}


output:

{
  "nowCalled": "2025-12-15T14:22:10.123+05:30[Asia/Kolkata]",
  "nowCalledSpecificTimeZone": "2025-12-15T03:52:10.123-05:00[America/New_York]"
}

example 2:


input:

import java.time.ZonedDateTime;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;

public class Main {
    public static void main(String[] args) {
        ZonedDateTime now = ZonedDateTime.now();

        long epochTime = now.toEpochSecond();
        int nanoseconds = now.getNano();
        int milliseconds = nanoseconds / 1_000_000;
        int seconds = now.getSecond();
        int minutes = now.getMinute();
        int hour = now.getHour();
        int day = now.getDayOfMonth();
        int month = now.getMonthValue();
        int year = now.getYear();
        int quarter = (month - 1) / 3 + 1;
        int dayOfWeek = now.getDayOfWeek().getValue(); // 1=Monday
        int dayOfYear = now.getDayOfYear();
        int offsetSeconds = now.getOffset().getTotalSeconds();

        String formattedDate = now.format(DateTimeFormatter.ofPattern("y-MM-dd"));
        String formattedTime = now.format(DateTimeFormatter.ofPattern("hh:m:s"));

        System.out.println("{");
        System.out.println("  \"now\": \"" + now + "\",");
        System.out.println("  \"epochTime\": " + epochTime + ",");
        System.out.println("  \"nanoseconds\": " + nanoseconds + ",");
        System.out.println("  \"milliseconds\": " + milliseconds + ",");
        System.out.println("  \"seconds\": " + seconds + ",");
        System.out.println("  \"minutes\": " + minutes + ",");
        System.out.println("  \"hour\": " + hour + ",");
        System.out.println("  \"day\": " + day + ",");
        System.out.println("  \"month\": " + month + ",");
        System.out.println("  \"year\": " + year + ",");
        System.out.println("  \"quarter\": " + quarter + ",");
        System.out.println("  \"dayOfWeek\": " + dayOfWeek + ",");
        System.out.println("  \"dayOfYear\": " + dayOfYear + ",");
        System.out.println("  \"offsetSeconds\": " + offsetSeconds + ",");
        System.out.println("  \"formattedDate\": \"" + formattedDate + "\",");
        System.out.println("  \"formattedTime\": \"" + formattedTime + "\"");
        System.out.println("}");
    }
}
