@GetMapping("/stringOps")
public Map<String, String> stringManipulationExample() {
    
    String myVar = "Hello World!";

    String indices2to6 = myVar.substring(2, 7); // "llo W"
    String indicesFromEnd = myVar.substring(6);  // "World!"
    String reversal = new StringBuilder(myVar).reverse().toString(); // "!dlroW olleH"

    return Map.of(
        "indices2to6", indices2to6,
        "indicesFromEnd", indicesFromEnd,
        "reversal", reversal
    );
}

output:

{
  "indices2to6": "llo W",
  "indicesFromEnd": "World!",
  "reversal": "!dlroW olleH"
}
