import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class IsDecimalExample {

    // Custom utility method to replicate DataWeave's isDecimal behavior
    public static boolean isDecimal(Object value) {
        if (value == null) {
            return false;
        } 
        
        // 1. Check if the value is a known Java floating-point type (Double or Float).
        if (value instanceof Double || value instanceof Float) {
            // DataWeave considers these true, even if the value is exactly X.0
            return true;
        }

        // 2. Check if the value is an Integer or Long (DataWeave returns false for these).
        if (value instanceof Integer || value instanceof Long) {
            return false;
        }
        
        // 3. Check if the value is a String representation of a decimal number.
        if (value instanceof String str) {
            // DataWeave's isDecimal("1.1") is true, but isDecimal("1") is false.
            // We check if the string contains a decimal point and can be parsed as a number.
            if (str.contains(".")) {
                try {
                    Double.parseDouble(str);
                    return true;
                } catch (NumberFormatException e) {
                    // String contains a dot but isn't a valid number (e.g., "1.a")
                    return false; 
                }
            }
            return false; // String without a decimal point ("1") is false
        }
        
        return false; // All other types (Boolean, Map, List, etc.) are false
    }

    public static void main(String[] args) {
        // --- Input Data ---
        // 1.1 (DataWeave Decimal/Number) -> Java Double
        Double input1 = 1.1; 
        // 1 (DataWeave Number/Integer) -> Java Integer
        Integer input2 = 1; 
        // "1.1" (DataWeave String) -> Java String
        String input3 = "1.1"; 

        // --- Perform Operations ---
        List<Boolean> results = Arrays.asList(
            isDecimal(input1),  // isDecimal(1.1) -> true
            isDecimal(input2),  // isDecimal(1) -> false
            isDecimal(input3)   // isDecimal("1.1") -> true
        );
        
        // --- Output ---
        System.out.println("Input Logic: Custom isDecimal(Object value)");
        System.out.println("Output (JSON equivalent): " + results);
    }
}
