    // --- Helper for DWL scan() logic ---
    private List<List<String>> scanMultiple(String input, String regex) {
        List<List<String>> allMatches = new ArrayList<>();
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(input);
        while (matcher.find()) {
            List<String> currentMatch = new ArrayList<>();
            currentMatch.add(matcher.group(0)); 
            for (int i = 1; i <= matcher.groupCount(); i++) {
                currentMatch.add(matcher.group(i));
            }
            allMatches.add(currentMatch);
        }
        return allMatches;
    }

    // --- Helper for DWL sizeOf() logic (for demonstration purposes only) ---
    private int sizeOf(Object value) {
        if (value instanceof List) return ((List<?>) value).size();
        if (value instanceof Map) return ((Map<?, ?>) value).size();
        if (value instanceof String) return ((String) value).length();
        if (value instanceof Number) return String.valueOf(value).length();
        if (value instanceof ZonedDateTime) return ((ZonedDateTime) value).toString().length();
        if (value instanceof LocalDateTime) return ((LocalDateTime) value).toString().length();
        // Fallback for DWL quirks like sizeOf("123" as Number) -> 1 or sizeOf(123) -> 1
        return 1; 
    }

    // --- Helper for DWL flatten(scan) logic ---
    private List<String> scanAndFlatten(String input, String regex) {
        List<String> results = new ArrayList<>();
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(input);
        if (matcher.find()) {
            results.add(matcher.group(0)); 
            for (int i = 1; i <= matcher.groupCount(); i++) {
                results.add(matcher.group(i));
            }
        }
        return results;
    }
