daysBetween:


input:

from("direct:daysBetween")
.setProperty("start",
    simple("${date:parse('2016-10-01T23:57:59-03:00','yyyy-MM-dd''T''HH:mm:ssXXX')}"))
.setProperty("end",
    simple("${date:parse('2017-10-01T23:57:59-03:00','yyyy-MM-dd''T''HH:mm:ssXXX')}"))

.setBody(simple(
    "{ \"days\": ${math:(${exchangeProperty.end} - ${exchangeProperty.start}) / 86400000} }"
));


output:


{ "days": 365 }
