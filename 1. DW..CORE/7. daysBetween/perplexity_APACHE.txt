import org.apache.camel.builder.RouteBuilder;
import java.time.*;
import java.time.temporal.ChronoUnit;

public class DaysBetweenRoute extends RouteBuilder {
    @Override
    public void configure() throws Exception {
        from("direct:daysBetween")
            .process(exchange -> {
                // Parse DataWeave timestamps
                ZonedDateTime start = ZonedDateTime.parse("2016-10-01T23:57:59-03:00");
                ZonedDateTime end = ZonedDateTime.parse("2017-10-01T23:57:59-03:00");
                
                // Calculate whole days (truncate to date for exact match)
                long days = ChronoUnit.DAYS.between(
                    start.toLocalDate(), 
                    end.toLocalDate()
                );
                
                exchange.getIn().setBody("{\"days\": " + days + "}");
            })
            .to("log:output?showAll=true");
    }
}
