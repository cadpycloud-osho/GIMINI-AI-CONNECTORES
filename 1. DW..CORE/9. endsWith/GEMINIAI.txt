package com.example.stringdemo;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;

@RestController
public class EndsWithController {

    /**
     * Replicates the DataWeave expression: [ "Mariano" endsWith "no", "Mariano" endsWith "to" ]
     * @return A list of booleans representing the result of the two checks.
     */
    @GetMapping("/checkEndsWith")
    public List<Boolean> checkEndsWith() {
        String baseString = "Mariano";

        // 1. Check if "Mariano" ends with "no" (Expected: true)
        boolean check1 = baseString.endsWith("no");

        // 2. Check if "Mariano" ends with "to" (Expected: false)
        boolean check2 = baseString.endsWith("to");

        // 3. Return the results as a list, which Spring Boot converts to a JSON array
        return Arrays.asList(check1, check2);
    }
}
