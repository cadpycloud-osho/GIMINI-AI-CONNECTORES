Alternative Stream Implementation

List<String> suffixes = Arrays.asList("no", "to");
List<Boolean> result = suffixes.stream()
    .map(suffix -> "Mariano".endsWith(suffix))
    .collect(Collectors.toList());

System.out.println(result);
