    public static void main(String[] args) {
        String[] arr = {"Bond", "James", "Bond"};

import java.util.regex.*;
import java.util.*;

public class Main {
    public static void main(String[] args) {
        String text = "I heart DataWeave";
        Pattern pattern = Pattern.compile("\\w*ea\\w*\\b");
        Matcher matcher = pattern.matcher(text);

        List<int[]> indices = new ArrayList<>();
        while (matcher.find()) {
            indices.add(new int[]{matcher.start(), matcher.end()});
        }

        // Print like DataWeave output
        for (int[] pair : indices) {
            System.out.println(Arrays.toString(pair));
        }
        // Output: [2, 7]
        //         [8, 17]
    }
}


output:
[[2, 7], [8, 17]]

import java.util.ArrayList;

public class Main {
    public static void main(String[] args) {
        String text = "aabccdbce";
        List<Integer> indices = new ArrayList<>();

        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == 'a') {
                indices.add(i);
            }
        }

        System.out.println(indices); // Output: [0, 1]
    }
}


output:
[0, 1]
