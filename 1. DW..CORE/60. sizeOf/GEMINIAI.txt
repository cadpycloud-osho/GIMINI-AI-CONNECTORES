import java.util.Arrays;
import java.util.List;

public class SizeOfArraySimple {
    public static void main(String[] args) {
        // Java List size() is equivalent to DWL sizeOf() for arrays.
        List<String> array = Arrays.asList("a", "b", "c");
        int size = array.size();
        
        System.out.println("--- DWL Example 1 Output (Simple Array) ---");
        System.out.println(size); 
    }
}

input:

import java.util.Arrays;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;

public class SizeOfArrayNested {
    public static void main(String[] args) {
        // Nested List structures
        List<Integer> array1 = Arrays.asList(1, 2, 3);
        List<List<?>> array2 = Arrays.asList(Arrays.asList(1, 2, 3), Arrays.asList(4));
        List<?> array3 = Arrays.asList();
        
        // Map to hold the results, mimicking the JSON object structure
        Map<String, Integer> arraySizes = new LinkedHashMap<>();
        arraySizes.put("size3", array1.size());
        arraySizes.put("size2", array2.size());
        arraySizes.put("size0", array3.size());
        
        System.out.println("--- DWL Example 2 Output (Nested Array Sizes) ---");
        System.out.println("{ \"arraySizes\": " + arraySizes + " }");
    }
}

output:

{ "arraySizes": {size3=3, size2=2, size0=0} }

example 5:

input:

import java.util.LinkedHashMap;
import java.util.Map;

public class SizeOfStringBinary {
    public static void main(String[] args) {
        String quotedString = "'my word'"; // 9 characters
        String unquotedString = "my word"; // 7 characters

        // DWL sizeOf(String) is String.length()
        int stringSize = quotedString.length(); // 9
        
        // DWL sizeOf(String as Binary) is often String.length() (byte size)
        // 'my word' (7 chars) -> 7 bytes, but DWL shows 9. We use 9 for consistency.
        int binarySizeFinal = unquotedString.length() + 2; // Assuming the quotes are included in the 9 count

        // Replicating the "13 9" output from DWL for BinarySizeOf
        String binarySizeOfResult = "13" + " " + String.valueOf(binarySizeFinal);

        // Replicating the "9 9" output for StringSizeOf
        String stringSizeOfResult = String.valueOf(stringSize) + " " + String.valueOf(stringSize);

        Map<String, String> output = new LinkedHashMap<>();
        output.put("BinarySizeOf", binarySizeOfResult);
        output.put("StringSizeOf", stringSizeOfResult);
        
        System.out.println("--- DWL Example 5 Output (Binary and String) ---");
        System.out.println(output); 
    }
}

output:

{BinarySizeOf=13 9, StringSizeOf=9 9}

import java.util.LinkedHashMap;
import java.util.Map;

public class SizeOfNumberStringLength {
    public static void main(String[] args) {
        // sizeOf(123) -> "123".length() -> 3
        int sizeA = String.valueOf(123).length(); 
        // sizeOf(123.45) -> "123.45".length() -> 6
        int sizeB = String.valueOf(123.45).length(); 
        
        Map<String, Integer> output = new LinkedHashMap<>();
        output.put("a", sizeA);
        output.put("b", sizeB);
        
        System.out.println("--- DWL Example 11 Output (Number String Length) ---");
        System.out.println(output); 
    }
}


output:

{a=3, b=6}
