example 2:


input:

import java.util.*;

public class Main {
    public static void main(String[] args) {

        Map<String, Integer> arraySizes = new HashMap<>();

        arraySizes.put("size3", Arrays.asList(1,2,3).size());
        arraySizes.put("size2", Arrays.asList(
                Arrays.asList(1,2,3),
                Arrays.asList(4)
        ).size());
        arraySizes.put("size0", Collections.emptyList().size());

        System.out.println(arraySizes);
    }
}


output:

{size3=3, size2=2, size0=0}

example 5:


input:

import java.nio.charset.StandardCharsets;

public class Main {
    public static void main(String[] args) {

        String value = "'my word'";

        int binarySize = value.getBytes(StandardCharsets.UTF_8).length;
        int stringSize = value.length();

        System.out.println("BinarySizeOf: " + binarySize);
        System.out.println("StringSizeOf: " + stringSize);
    }
}


output:

BinarySizeOf: 13
StringSizeOf: 9
