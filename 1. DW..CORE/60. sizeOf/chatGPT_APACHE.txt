sizeOf:

example 1:

input:
from("direct:start")
    .process(exchange -> {
        int size = java.util.Arrays.asList("a","b","c").size();
        exchange.getIn().setBody(size);
    });



output:

3



-----------------------------------------------------------

example 2:

input:

from("direct:start")
    .process(exchange -> {
        java.util.Map<String,Object> sizes = new java.util.LinkedHashMap<>();
        sizes.put("size3", java.util.Arrays.asList(1,2,3).size());
        sizes.put("size2", java.util.Arrays.asList(
                java.util.Arrays.asList(1,2,3),
                java.util.Arrays.asList(4)
        ).size());
        sizes.put("size0", java.util.Collections.emptyList().size());
        exchange.getIn().setBody(java.util.Collections.singletonMap("arraySizes", sizes));
    });


output:

{
  "arraySizes": {
    "size3": 3,
    "size2": 2,
    "size0": 0
  }
}

example 5:

input:

    .process(exchange -> {
        byte[] binary = "'my word'".getBytes();
        String text = "'my word'";
        java.util.Map<String,String> result = new java.util.LinkedHashMap<>();
        result.put("BinarySizeOf", binary.length + " " + binary.length);
        result.put("StringSizeOf", text.length() + " " + text.length());
        exchange.getIn().setBody(result);
    });


output:

{
  "BinarySizeOf": "9 9",
  "StringSizeOf": "9 9"
