@RestController
public class SimpleController {

    @Value("${http.port}")
    private String httpPort;

    private static final Logger logger = LoggerFactory.getLogger(SimpleController.class);

    @GetMapping("/log-port")
    public String logPort() {
        logger.info("{}", httpPort);
        return "Logged: " + httpPort;
    }
}
