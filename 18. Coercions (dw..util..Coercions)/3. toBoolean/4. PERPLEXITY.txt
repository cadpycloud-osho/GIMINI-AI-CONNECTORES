package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class BooleanCoercionsController {
    
    @GetMapping("/boolean-coercions")
    public BooleanResponse getBooleanCoercions() {
        return new BooleanResponse(
            parseBoolean("true"),
            parseBoolean("false"),
            parseBoolean("FALSE"),
            parseBoolean("TrUe")
        );
    }
    
    private boolean parseBoolean(String value) {
        return Boolean.parseBoolean(value.toLowerCase());
    }
    
    public static class BooleanResponse {
        public boolean a, b, c, d;
        
        public BooleanResponse(boolean a, boolean b, boolean c, boolean d) {
            this.a = a;
            this.b = b;
            this.c = c;
            this.d = d;
        }
    }
}
