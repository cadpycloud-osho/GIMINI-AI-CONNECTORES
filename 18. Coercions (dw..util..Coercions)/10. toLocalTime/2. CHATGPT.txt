input:

package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.time.LocalTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.HashMap;
import java.util.Map;

@RestController
public class LocalTimeCoercionController {

    @GetMapping("/localtime")
    public Map<String, String> getLocalTime() {

        Map<String, String> result = new HashMap<>();

        // a: successful conversion using second formatter
        result.put("a", toLocalTime(
                "23:57:59",
                new String[]{
                        "HH:mm:ss.n",
                        "HH:mm:ss"
                }
        ));

        // b: error message when formatter list fails
        result.put("b", tryToLocalTime(
                "23:57:59",
                new String[]{
                        "HH:mm:ss.n"
                }
        ));

        return result;
    }

    private String toLocalTime(String value, String[] formats) {
        for (String format : formats) {
            try {
                DateTimeFormatter formatter = DateTimeFormatter.ofPattern(format);
                LocalTime time = LocalTime.parse(value, formatter);
                return time.toString(); // HH:mm:ss
            } catch (DateTimeParseException ignored) {
            }
        }
        throw new RuntimeException("Could not find a valid formatter for '" + value + "'");
    }

    private String tryToLocalTime(String value, String[] formats) {
        try {
            return toLocalTime(value, formats);
        } catch (RuntimeException e) {
            return e.getMessage();
        }
    }
}


output:

{
  "a": "23:57:59",
  "b": "Could not find a valid formatter for '23:57:59'"
}

example 2:


input:

package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.time.LocalTime;
import java.time.format.DateTimeFormatter;
import java.util.HashMap;
import java.util.Map;

@RestController
public class LocalTimeController {

    @GetMapping("/localtime")
    public Map<String, String> getLocalTimes() {

        Map<String, String> result = new HashMap<>();

        // Equivalent to toLocalTime("23:57:59")
        result.put("toLocalTimeEx", toLocalTime("23:57:59", null));

        // Equivalent to toLocalTime("13:44:12.283","HH:mm:ss.n")
        result.put("toLocalTimeEx2", toLocalTime("13:44:12.283", "HH:mm:ss.SSS"));

        return result;
    }

    private String toLocalTime(String value, String pattern) {
        LocalTime time;
        if (pattern != null) {
            DateTimeFormatter formatter = DateTimeFormatter.ofPattern(pattern);
            time = LocalTime.parse(value, formatter);
        } else {
            time = LocalTime.parse(value); // default ISO_LOCAL_TIME
        }
        return time.toString();
    }
}


output:

{
  "toLocalTimeEx": "23:57:59",
  "toLocalTimeEx2": "13:44:12.283"
}
