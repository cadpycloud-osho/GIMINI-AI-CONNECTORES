package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.time.OffsetTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.HashMap;
import java.util.Map;

@RestController
public class TimeCoercionController {

    @GetMapping("/timeOrNull")
    public Map<String, String> getTimeOrNull() {

        String input = "13:44:12.283-08:00";

        Map<String, String> result = new HashMap<>();
        result.put("a", toTimeOrNull(input, new String[]{
                "HH:mm:ss.SSS",
                "HH:mm:ss.nXXX" // DW's "nxxx" â‰ˆ nanoseconds with offset
        }));

        result.put("b", toTimeOrNull(input, new String[]{
                "HH:mm:ss.SSS"
        }));

        return result;
    }

    private String toTimeOrNull(String value, String[] formats) {
        for (String format : formats) {
            try {
                DateTimeFormatter formatter = DateTimeFormatter.ofPattern(format);
                OffsetTime time = OffsetTime.parse(value, formatter);
                return time.toString(); // ISO-8601 time string
            } catch (DateTimeParseException ignored) {
            }
        }
        return null;
    }
}
