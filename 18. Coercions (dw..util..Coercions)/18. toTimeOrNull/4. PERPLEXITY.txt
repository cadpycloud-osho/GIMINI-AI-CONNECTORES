@SpringBootApplication
@RestController
import java.time.OffsetTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeFormatterBuilder;
import java.time.format.DateTimeParseException;
import java.time.temporal.ChronoField;
import java.util.Map;

public class TimeOrNullApp {

    public static void main(String[] args) {
        SpringApplication.run(TimeOrNullApp.class, args);
    }

    @GetMapping("/time-or-null")
    public Map<String, Object> timeOrNull() {
        String input = "13:44:12.283-08:00";
        
        // Equivalent to DataWeave 'a' - multiple format fallback
        OffsetTime resultA = parseWithMultipleFormats(input);
        
        // Equivalent to DataWeave 'b' - single format only
        OffsetTime resultB = parseWithSingleFormat(input);
        
        return Map.of(
            "a", resultA,
            "b", resultB
        );
    }
    
    private OffsetTime parseWithMultipleFormats(String input) {
        // First format: HH:mm:ss.xxx (milliseconds - expects exactly 3 digits)
        DateTimeFormatter fmt1 = DateTimeFormatter.ofPattern("HH:mm:ss.SSSXXX");
        try {
            return OffsetTime.parse(input, fmt1);
        } catch (DateTimeParseException e) {
            // Fallback: HH:mm:ss.nxxx (nanoseconds - flexible 1-9 digits)
            DateTimeFormatter fmt2 = new DateTimeFormatterBuilder()
                .appendPattern("HH:mm:ss.")
                .appendFraction(ChronoField.NANO_OF_SECOND, 1, 9, true)
                .appendOffset("+HH:mm", "Z")
                .toFormatter();
            try {
                return OffsetTime.parse(input, fmt2);
            } catch (DateTimeParseException ex) {
                return null;
            }
        }
    }
    
    private OffsetTime parseWithSingleFormat(String input) {
        try {
            DateTimeFormatter fmt = DateTimeFormatter.ofPattern("HH:mm:ss.SSSXXX");
            return OffsetTime.parse(input, fmt);
        } catch (DateTimeParseException e) {
            return null; // DataWeave toTimeOrNull behavior
        }
    }
}
