import java.util.Map;

    public static void main(String[] args) {

        Map<String, String> result = new HashMap<>();

        // Equivalent of DW toTimeZone("-03:00")

        // Equivalent of DW toTimeZone("Z")
        result.put("toTimeZoneAbbreviation", ZoneOffset.of("Z").toString());

        // Equivalent of DW toTimeZone("America/Argentina/Buenos_Aires")

        System.out.println(result);
    }
}
