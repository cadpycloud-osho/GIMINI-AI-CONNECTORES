    /**
     * Mimics DataWeave's toLocalDateTimeOrNull() behavior with chained formatters.
     * Returns the parsed LocalDateTime upon success, or null if all formatters fail.
     */
    private LocalDateTime toLocalDateTimeOrNull(String dateTimeString, DateTimeFormatter... formatters) {
        for (DateTimeFormatter formatter : formatters) {
            try {
                // Attempt to parse with the current formatter
                return LocalDateTime.parse(dateTimeString, formatter);
            } catch (DateTimeParseException e) {
                // DataWeave's behavior: swallow the exception and continue to the next formatter
            }
        }
        // If the loop completes without a successful parse, return null
        return null; 
    }
