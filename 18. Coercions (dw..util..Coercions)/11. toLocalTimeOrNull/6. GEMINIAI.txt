public class DataWeaveToLocalTimeOrNullConverter {

    /**
     * Mimics DataWeave's toLocalTimeOrNull() behavior with chained formatters.

    private LocalTime toLocalTimeOrNull(String timeString, DateTimeFormatter... formatters) {
        for (DateTimeFormatter formatter : formatters) {
            try {
                // Attempt to parse with the current formatter
                return LocalTime.parse(timeString, formatter);

                // DataWeave's behavior: swallow the exception and continue to the next formatter
            }
        }
        // If the loop completes without a successful parse, return null
        return null; 
    }
