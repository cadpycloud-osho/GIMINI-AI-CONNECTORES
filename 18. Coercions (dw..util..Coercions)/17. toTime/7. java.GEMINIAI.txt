            try {
                return OffsetTime.parse(timeString, formatter);
            } catch (DateTimeParseException e) {
                // Ignore the error and try the next formatter
            }
        }
        return null; // Return null if all parsing attempts fail

import java.time.OffsetTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeFormatterBuilder;
import java.util.HashMap;
import java.util.Locale;
import java.util.Map;

public class Main {

    public Map<String, String> getTimeConversionMap() {
        
        // --- Logic for Key 'a' ---
        // Input time string is in the standard ISO 8601 format (including 'Z' for ZoneOffset.UTC)
        String timeStringA = "23:57:59Z";
        // OffsetTime.parse() uses ISO_OFFSET_TIME by default
        OffsetTime parsedTimeA = OffsetTime.parse(timeStringA); 

        // --- Logic for Key 'b' ---
        String timeStringB = "13:44:12.283-08:00";
        
        // Custom formatter to handle the fractional seconds:
        // .appendFraction(ChronoField.NANO_OF_SECOND, 1, 9, true) handles 1 to 9 digits for nanoseconds.
        // It correctly handles the input "12.283" (3 digits of precision).
        DateTimeFormatter customFormatter = new DateTimeFormatterBuilder()
            .appendPattern("HH:mm:ss")
            .appendFraction(java.time.temporal.ChronoField.NANO_OF_SECOND, 1, 9, true)
            .appendPattern("XXX") // Offset format, e.g., -08:00
            .toFormatter(Locale.US);
            
        OffsetTime parsedTimeB = OffsetTime.parse(timeStringB, customFormatter);

        // Create the root map
        Map<String, String> rootMap = new HashMap<>();
        
        // Add the results (OffsetTime.toString() provides the ISO format, 
        // including full nanosecond padding for fractional seconds when present.)
        rootMap.put("a", parsedTimeA.toString());
        // For 'b', the parsed value 13:44:12.283-08:00 is represented internally with nanosecond precision. 
        // OffsetTime.toString() pads the fraction up to 9 digits if needed (283000000 nanoseconds).
        // However, the requested output format is very specific. I'll rely on the standard Java 
        // output for the value 13:44:12.283-08:00. DataWeave typically uses a smaller precision in its default string representation.
        // The expected output b=13:44:12.000000283-08:00 is mathematically incorrect for the input 13:44:12.283-08:00.
        // The correct nanosecond-precise representation of .283 is 13:44:12.283000000-08:00.
        // I will use the actual Java output, but note the discrepancy.
        rootMap.put("b", parsedTimeB.toString());

        return rootMap;
    }

    public static void main(String[] args) {
        // FIX: Instantiate the Main class, which contains the logic.
        Main converter = new Main();
        Map<String, String> result = converter.getTimeConversionMap();
        
        // Print the Java structure, matching the requested output format.
        System.out.println("Result: " + result);
    }
}
