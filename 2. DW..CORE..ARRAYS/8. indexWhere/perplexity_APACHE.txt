import org.apache.camel.builder.RouteBuilder;
import java.util.Arrays;
import java.util.List;

public class IndexWhereRoute extends RouteBuilder {
    @Override
    public void configure() throws Exception {
        from("direct:indexWhere")
            .process(exchange -> {
                List<String> users = Arrays.asList("Mariano", "Leandro", "Julian");
                
                // indexWhere (item) -> item startsWith "Jul"
                int index = IntStream.range(0, users.size())
                    .filter(i -> users.get(i).startsWith("Jul"))
                    .findFirst()
                    .orElse(-1);
                
                exchange.getIn().setBody(index);
            })
            .to("log:output?showAll=true");
    }
}
