import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;
import java.util.function.Predicate;

@SpringBootApplication
public class IndexWhereDemoApplication {
    public static void main(String[] args) {
        SpringApplication.run(IndexWhereDemoApplication.class, args);
    }
}

@RestController
class IndexWhereController {
    
    private <T> int indexWhere(List<T> list, Predicate<T> predicate) {
        for (int i = 0; i < list.size(); i++) {
            if (predicate.test(list.get(i))) {
                return i;
            }
        }
        return -1;
    }
    
    @GetMapping("/indexWhere")
    public int indexWhereExample() {
        List<String> users = Arrays.asList("Mariano", "Leandro", "Julian");
        return indexWhere(users, item -> item.startsWith("Jul"));
    }
}
