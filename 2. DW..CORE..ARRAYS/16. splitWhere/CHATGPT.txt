package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.*;

@RestController
public class SplitWhereController {

    @GetMapping("/splitWhere")
    public Map<String, List<String>> splitWhereExample() {

        List<String> users = Arrays.asList("Mariano", "Leandro", "Julian", "Tomo");
        
        int index = -1;

        // find index of first element starting with "Jul"
        for (int i = 0; i < users.size(); i++) {
            if (users.get(i).startsWith("Jul")) {
                index = i;
                break;
            }
        }

        List<String> left;
        List<String> right;

        if (index >= 0) {
            left = users.subList(0, index);
            right = users.subList(index, users.size());
        } else {
            left = new ArrayList<>(users);
            right = Collections.emptyList();
        }

        Map<String, List<String>> response = new HashMap<>();
        response.put("l", left);
        response.put("r", right);

        return response;
    }
}
