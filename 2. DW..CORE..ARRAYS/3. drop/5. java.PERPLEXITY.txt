public class DropArrayExample {
    public static void main(String[] args) {
        String[] users = {"Mariano", "Leandro", "Julian"};
        int n = 2; // number to drop

        int newSize = Math.max(0, users.length - n);
        String[] result = new String[newSize];

        for (int i = 0; i < newSize; i++) {
            result[i] = users[i + n];
        }

        // print result
        System.out.print("[");
        for (int i = 0; i < result.length; i++) {
            System.out.print(result[i]);
            if (i < result.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}

[Julian]
