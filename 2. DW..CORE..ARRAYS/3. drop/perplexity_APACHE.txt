import org.apache.camel.builder.RouteBuilder;
import java.util.Arrays;
import java.util.List;

public class DropRoute extends RouteBuilder {
    @Override
    public void configure() throws Exception {
        from("direct:drop")
            .process(exchange -> {
                List<String> users = Arrays.asList("Mariano", "Leandro", "Julian");
                
                // drop(users, 2) → skip first 2 elements → ["Julian"]
                List<String> result = users.subList(2, users.size());
                
                exchange.getIn().setBody(result);
            })
            .marshal().json(JsonLibrary.Jackson)
            .to("log:output?showAll=true");
    }
}
