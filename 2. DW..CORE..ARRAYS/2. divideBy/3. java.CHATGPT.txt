import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class Main {

    // Utility method similar to DataWeave divideBy
    public static <T> List<List<T>> divideBy(List<T> list, int size) {
        List<List<T>> result = new ArrayList<>();

        for (int i = 0; i < list.size(); i += size) {
            result.add(list.subList(i, Math.min(i + size, list.size())));
        }
        return result;
    }

    public static void main(String[] args) {

        List<Integer> list1 = Arrays.asList(1, 2, 3, 4, 5);
        List<Integer> list2 = Arrays.asList(1, 2, 3, 4, 5, 6);

        List<List<Integer>> divideBy2_1 = divideBy(list1, 2);
        List<List<Integer>> divideBy2_2 = divideBy(list2, 2);
        List<List<Integer>> divideBy3   = divideBy(list1, 3);

        System.out.println("{");
        System.out.println(" \"divideBy\": [");

        System.out.println("  { \"divideBy2\": " + divideBy2_1 + " },");
        System.out.println("  { \"divideBy2\": " + divideBy2_2 + " },");
        System.out.println("  { \"divideBy3\": " + divideBy3   + " }");

        System.out.println(" ]");
        System.out.println("}");
    }
