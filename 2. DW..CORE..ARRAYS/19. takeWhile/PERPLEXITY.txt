import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

@SpringBootApplication
public class TakeWhileDemoApplication {
    public static void main(String[] args) {
        SpringApplication.run(TakeWhileDemoApplication.class, args);
    }
}

@RestController
class TakeWhileController {
    
    private <T> List<T> takeWhile(List<T> list, java.util.function.Predicate<T> predicate) {
        int takeCount = 0;
        while (takeCount < list.size() && predicate.test(list.get(takeCount))) {
            takeCount++;
        }
        return list.subList(0, takeCount);
    }
    
    @GetMapping("/takeWhile")
    public List<Integer> takeWhileExample() {
        List<Integer> arr = Arrays.asList(0, 1, 2, 1);
        return takeWhile(arr, x -> x <= 1);
    }
}
