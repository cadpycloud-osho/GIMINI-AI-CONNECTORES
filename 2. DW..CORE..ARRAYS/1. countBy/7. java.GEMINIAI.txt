import java.util.Arrays;
import java.util.List;

public class Main {

    public static void main(String[] args) {
        // Source data array
        List<Integer> list = Arrays.asList(1, 2, 3, 4);
        
        int countOfEvenNumbers = 0;

        // DataWeave Logic: countBy (($ mod 2) == 0)
        // Loop through each number and check if it's even
        for (int number : list) {
            if ((number % 2) == 0) {
                countOfEvenNumbers++; 
            }
        }
        
        // The final result structure: {"countBy": 2}
        // Outputting the required JSON-like map structure
        System.out.println("{\"countBy\":" + countOfEvenNumbers + "}");
    }
}
