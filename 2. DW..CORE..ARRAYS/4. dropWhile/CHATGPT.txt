package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

@RestController
public class DropWhileController {

    @GetMapping("/dropWhile")
    public List<Integer> dropWhileExample() {

        List<Integer> arr = new ArrayList<>(Arrays.asList(0, 1, 3, 2, 1));
        int index = 0;

        while (index < arr.size() && arr.get(index) < 3) {
            index++;
        }

        return arr.subList(index, arr.size());
    }
}
