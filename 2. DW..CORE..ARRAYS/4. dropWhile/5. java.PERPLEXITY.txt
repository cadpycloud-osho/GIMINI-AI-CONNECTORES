import java.util.List;

public class Main {
    public static void main(String[] args) {
        int[] arr = {0, 1, 3, 2, 1};
        int i = 0;
        
        // Drop while condition is true (element < 3)

Java Streams Version
This functional approach uses dropWhile-like logic with Streams, finding the first matching index then slicing.

java
import java.util.Arrays;
import java.util.IntSummaryStatistics;
import java.util.stream.IntStream;

public class DropWhileStream {
    public static void main(String[] args) {
        int[] arr = {0, 1, 3, 2, 1};
        
        // Find first index where condition fails (< 3)
        int dropCount = IntStream.range(0, arr.length)
                                .filter(i -> arr[i] >= 3)
                                .findFirst()
                                .orElse(arr.length);
        
        // Get remaining elements
        int[] result = Arrays.copyOfRange(arr, dropCount, arr.length);
        System.out.println(Arrays.toString(result));
    }
}
Output: [3, 2, 1]

Manual Array Version (No Streams)
Pure array manipulation without Java 8+ features, matching basic Java style.

java
public class DropWhileManual {
    public static void main(String[] args) {
        int[] arr = {0, 1, 3, 2, 1};
        int startIndex = 0;
        
        // Find first element >= 3
        while (startIndex < arr.length && arr[startIndex] < 3) {
            startIndex++;
        }
        
        // Print remaining elements as JSON array
        System.out.print("[");
        for (int i = startIndex; i < arr.length; i++) {
            System.out.print(arr[i]);
            if (i < arr.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
}
Output: [3, 2, 1]
