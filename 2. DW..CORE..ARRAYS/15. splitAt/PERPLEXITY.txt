import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.HashMap;

@SpringBootApplication
public class SplitAtDemoApplication {
    public static void main(String[] args) {
        SpringApplication.run(SplitAtDemoApplication.class, args);
    }
}

@RestController
class SplitAtController {
    
    @GetMapping("/splitAt")
    public Map<String, List<String>> splitAtExample() {
        List<String> users = Arrays.asList("Mariano", "Leandro", "Julian");
        int splitIndex = 1;
        
        Map<String, List<String>> result = new HashMap<>();
        result.put("l", users.subList(0, splitIndex));
        result.put("r", users.subList(splitIndex, users.size()));
        
        return result;
    }
}
