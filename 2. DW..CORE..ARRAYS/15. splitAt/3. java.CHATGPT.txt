public class Main {

        int splitIndex = 1;

        List<String> right = new ArrayList<>(users.subList(splitIndex, users.size()));

        Map<String, List<String>> result = new LinkedHashMap<>();
        result.put("l", left);
        result.put("r", right);

        System.out.println(result);
    }
}
