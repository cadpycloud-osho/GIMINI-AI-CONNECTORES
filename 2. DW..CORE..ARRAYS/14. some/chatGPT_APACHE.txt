from("direct:someExample")
.process(exchange -> {

    Map<String, Object> results = new HashMap<>();

    // "ok" group
    List<Boolean> ok = new ArrayList<>();
    ok.add(List.of(1,2,3).stream().anyMatch(n -> n % 2 == 0));
    ok.add(List.of(1,2,3).stream().anyMatch(nextNum -> nextNum % 2 == 0));
    ok.add(List.of(1,2,3).stream().anyMatch(n -> n % 2 == 1));
    ok.add(List.of(1,2,3,4,5,6,7,8).stream().anyMatch(n -> {
        System.out.println("should stop at 2 ==" + n);
        return n == 2;
    }));
    ok.add(List.of(1,2,3).stream().anyMatch(n -> n == 1));
    ok.add(List.of(1,1,1).stream().anyMatch(n -> n == 1));
    ok.add(List.of(1).stream().anyMatch(n -> n == 1));

    // "err" group
    List<Boolean> err = new ArrayList<>();
    err.add(List.of(1,2,3).stream().anyMatch(n -> n == 100));
    err.add(List.of(1).stream().anyMatch(n -> n == 2));

    // Wrap into result map
    results.put("results", List.of(
        Map.of("ok", ok),
        Map.of("err", err)
    ));

    exchange.getIn().setBody(results);
});
