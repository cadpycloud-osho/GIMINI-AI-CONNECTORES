import java.util.LinkedHashMap;
import java.util.Map;

public class Main {

    public static String rightPad(String input, int length) {
        if (input == null) return null;
        if (length <= input.length()) return input;

        int paddingLength = length - input.length();
        StringBuilder sb = new StringBuilder(input);

        for (int i = 0; i < paddingLength; i++) {
            sb.append(' ');
        }

        return sb.toString();
    }

    public static void main(String[] args) {

        Map<String, String> result = new LinkedHashMap<>();

        result.put("a", rightPad(null, 3));
        result.put("b", rightPad("", 3));
        result.put("c", rightPad("bat", 5));
        result.put("d", rightPad("bat", 3));
        result.put("e", rightPad("bat", -1));

        System.out.println(result);
    }
}
