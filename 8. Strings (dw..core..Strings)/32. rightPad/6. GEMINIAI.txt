public class PadUtil {

    /**
     * Replicates the dw::core::Strings::rightPad function using the default space character.
     *
     * @param str The input string (can be null).
     * @param size The minimum desired length.
     * @return The padded string, or null if the input string was null.
     */
    public static String rightPad(String str, int size) {
        // 1. DataWeave Rule: If the input string is null, return null.
        if (str == null) {
            return null;
        }

        // 2. If the size is less than or equal to the current length, return the original string.
        if (size <= str.length()) {
            return str;
        }

        // 3. Calculate padding length and append spaces.
        int pads = size - str.length();
        StringBuilder sb = new StringBuilder(str);
        
        // Create the padding string
        // In a real application, StringUtils.repeat(' ', pads) would be used.
        for (int i = 0; i < pads; i++) {
            sb.append(' '); 
        }

        return sb.toString();
    }
}
