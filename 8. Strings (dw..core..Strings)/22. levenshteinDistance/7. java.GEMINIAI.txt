public class Main {

    /**
     * Static method to emulate DataWeave's levenshteinDistance(text1, text2) function.
     * Calculates Levenshtein distance using Dynamic Programming.
     */
    public static int calculateLevenshteinDistance(String s1, String s2) {
        if (s1 == null) return s2 == null ? 0 : s2.length();
        if (s2 == null) return s1.length();

        int m = s1.length();
        int n = s2.length();
        
        // Dynamic programming matrix initialization
        int[][] d = new int[m + 1][n + 1];

        // Initialize boundary conditions
        for (int i = 0; i <= m; i++) {
            d[i][0] = i; 
        }
        for (int j = 0; j <= n; j++) {
            d[0][j] = j; 
        }

        // Fill the matrix
        for (int j = 1; j <= n; j++) {
            for (int i = 1; i <= m; i++) {
                int cost = s1.charAt(i - 1) == s2.charAt(j - 1) ? 0 : 1;
                
                // Minimum of deletion, insertion, or substitution
                d[i][j] = Math.min(
                    Math.min(d[i - 1][j] + 1, d[i][j - 1] + 1), 
                    d[i - 1][j - 1] + cost
                );
            }
        }
        return d[m][n];
    }

    public static void main(String[] args) {
        String s1 = "kitten"; 
        String s2 = "sitting";
        
        int distance = calculateLevenshteinDistance(s1, s2);
            
        System.out.println("DataWeave");
        System.out.println("Output");
        
        // Print the result to match the DataWeave output structure
        System.out.println(distance);
    }
