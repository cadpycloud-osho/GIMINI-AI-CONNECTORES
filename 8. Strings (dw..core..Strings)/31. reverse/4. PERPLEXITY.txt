@RestController
@RequestMapping("/api")
public class ReverseController {
    
    @GetMapping("/reverse")
    public Map<String, Object> reverseExamples() {
        return Map.of(
            "a", reverse("Mariano"),
            "b", reverse(null),
            "c", reverse("")
        );
    }
    
    private String reverse(String input) {
        if (input == null || input.isEmpty()) {
            return input;
        }
        return new StringBuilder(input).reverse().toString();
    }
}
