import java.util.*;

public class DasherizeExample {

    public static void main(String[] args) {

        Map<String, String> output = new LinkedHashMap<>();

        output.put("a", dasherize("customer"));
        output.put("b", dasherize("customer_first_name"));
        output.put("c", dasherize("customer NAME"));
        output.put("d", dasherize("customerName"));

        System.out.println(output);
    }

    private static String dasherize(String input) {
        if (input == null || input.isEmpty()) return input;

        // 1. Replace underscores and spaces with dashes
        String normalized = input.replace("_", "-").replace(" ", "-");

        // 2. Insert dash before uppercase letters (except the first character)
        StringBuilder result = new StringBuilder();
        char[] chars = normalized.toCharArray();

        for (int i = 0; i < chars.length; i++) {
            char c = chars[i];
            if (Character.isUpperCase(c) && i > 0 && chars[i - 1] != '-') {
                result.append('-').append(Character.toLowerCase(c));
            } else {
                result.append(Character.toLowerCase(c));
            }
        }

        return result.toString();
    }
}
