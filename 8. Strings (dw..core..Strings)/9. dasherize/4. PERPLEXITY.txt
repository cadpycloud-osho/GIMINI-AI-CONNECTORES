package com.example.demo.controller;

import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.regex.Pattern;

@RestController
public class DasherizeController {
    
    @GetMapping(value = "/dasherize", produces = MediaType.APPLICATION_JSON_VALUE)
    public DasherizeResponse dasherize() {
        return new DasherizeResponse(
            dasherize("customer"),
            dasherize("customer_first_name"),
            dasherize("customer NAME"),
            dasherize("customerName")
        );
    }
    
    private String dasherize(String input) {
        // Replace underscores, spaces, and camelCase boundaries with hyphens
        return input.replaceAll("([a-z])([A-Z])", "$1-$2")
                   .replaceAll("_", "-")
                   .replaceAll("\\s+", "-")
                   .toLowerCase();
    }
    
    public static class DasherizeResponse {
        public final String a, b, c, d;
        
        public DasherizeResponse(String a, String b, String c, String d) {
            this.a = a; this.b = b; this.c = c; this.d = d;
        }
    }
}
