import java.util.*;

public class SubstringEveryConverter {
    public static List<String> substringEvery(String text, int length) {
        if (text == null || text.isEmpty() || length <= 0) {
            return new ArrayList<>();
        }
        
        List<String> result = new ArrayList<>();
        for (int i = 0; i < text.length(); i += length) {
            int end = Math.min(i + length, text.length());
            result.add(text.substring(i, end));
        }
        return result;
    }
    
    public static void main(String[] args) {
        List<String> result = substringEvery("substringEvery", 3);
        
        Map<String, Object> output = new LinkedHashMap<>();
        output.put("substringEvery", result);
        System.out.println(output);
    }
}
