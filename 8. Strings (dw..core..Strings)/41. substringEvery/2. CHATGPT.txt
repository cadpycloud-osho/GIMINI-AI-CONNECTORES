package com.example.demo.util;

import java.util.ArrayList;
import java.util.List;

public class StringUtils {

    public static List<String> substringEveryDW(String input, int n) {
        List<String> result = new ArrayList<>();
        if (input == null || n <= 0) return result;

        int length = input.length();
        for (int i = 0; i < length; i += n) {
            int end = Math.min(i + n, length);
            result.add(input.substring(i, end));
        }
        return result;
    }
}

package com.example.demo.controller;

import com.example.demo.util.StringUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.List;

@RestController
public class SubstringEveryController {

    @GetMapping("/substringevery")
    public List<String> substringEveryExample() {
        String input = "substringEvery";
        return StringUtils.substringEveryDW(input, 3);
    }
