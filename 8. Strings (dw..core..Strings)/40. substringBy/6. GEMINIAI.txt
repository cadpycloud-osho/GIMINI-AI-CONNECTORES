import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.Arrays;
import java.util.List;
import java.util.Map;

@RestController
public class SplitByController {

    @GetMapping("/split-by-conversion")
    public Map<String, List<String>> convertDataWeaveSplitBy() {
        
        String inputString = "hello~world=here_data-weave";
        
        // --- Core DataWeave Logic Equivalent ---
        // The condition $ == "~" or $ == "=" or $ == "_" translates to the regex [~=_].
        // Java's String.split() uses regex.
        String regexSeparator = "[~=_]"; 
        
        String[] parts = inputString.split(regexSeparator);
        List<String> result = Arrays.asList(parts);
        // ------------------------------------
        
        // Return a Map, which Spring Boot/Jackson will serialize to the required JSON list
        return Map.of(
            "result", result
        );
    }
}
