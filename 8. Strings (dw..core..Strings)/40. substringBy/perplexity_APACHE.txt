import java.util.*;

public class SubstringByConverter {
    public static List<String> substringBy(String text, java.util.function.Predicate<String> predicate) {
        if (text == null || text.isEmpty()) return new ArrayList<>();
        
        List<String> result = new ArrayList<>();
        StringBuilder current = new StringBuilder();
        
        for (int i = 0; i < text.length(); i++) {
            String ch = String.valueOf(text.charAt(i));
            if (predicate.test(ch)) {
                // Split point found - add current segment (non-empty only)
                if (current.length() > 0) {
                    result.add(current.toString());
                    current = new StringBuilder();
                }
            } else {
                current.append(ch);
            }
        }
        
        // Add final segment if non-empty
        if (current.length() > 0) {
            result.add(current.toString());
        }
        
        return result;
    }
    
    public static void main(String[] args) {
        String input = "hello~world=here_data-weave";
        List<String> result = substringBy(input, ch -> ch.equals("~") || ch.equals("=") || ch.equals("_"));
        
        Map<String, Object> output = new LinkedHashMap<>();
        output.put("substringBy", result);
        System.out.println(output);
    }
}
