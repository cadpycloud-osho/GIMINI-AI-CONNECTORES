public class Main {

    public static boolean isLowerCaseDW(String input) {
        if (input == null || input.isEmpty()) return false;

        // Unicode lowercase check — no digits, spaces, symbols
        if (!input.matches("^[\\p{Ll}]+$")) {
            return false;
        }

        return true;
    }
