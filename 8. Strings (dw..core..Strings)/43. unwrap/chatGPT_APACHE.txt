        result.put("c", unwrap("'abc'", "'"));
        result.put("d", unwrap("AABabcBAA", "A"));

        result.put("f", unwrap("#A", "#"));
        result.put("g", unwrap("A#", "#"));

        System.out.println(result);

    public static String unwrap(String value, String charToRemove) {
        if (value == null) {
            return null;
        }
        if (charToRemove == null || charToRemove.isEmpty()) {
            return value;

        int start = 0;
        int end = value.length();

            start = 1;
        }
