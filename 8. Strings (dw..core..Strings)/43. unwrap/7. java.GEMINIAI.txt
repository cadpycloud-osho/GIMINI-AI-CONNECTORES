    // The utility method is included here for completeness.
    public static class StringUnwrapUtil {
        public static String unwrap(String str, String wrapper) {
            if (str == null) {
                return null;
            }
            if (wrapper == null || wrapper.isEmpty()) {
                return str;
            }

            if (str.startsWith(wrapper) && str.endsWith(wrapper) && str.length() >= 2 * wrapper.length()) {
                // Remove prefix and suffix wrapper
                return str.substring(wrapper.length(), str.length() - wrapper.length());
            }

            return str;
        }
    }

    /**
     * Helper to format the Java String result into a valid JSON string (with quotes)
     * or the JSON literal 'null'.
     */
    private static String formatJsonString(String value) {
        // If the Java String is null, return the JSON literal "null". Otherwise, quote the string value.
        return value == null ? "null" : "\"" + value + "\"";
    }
