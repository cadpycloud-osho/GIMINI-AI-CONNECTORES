    public static int calculateHammingDistance(String s1, String s2) {
        if (s1 == null || s2 == null || s1.length() != s2.length()) {
            // As per definition, length must be equal.
            // We can throw an IllegalArgumentException for unequal lengths.
            throw new IllegalArgumentException("Cannot calculate Hamming distance for strings of unequal length.");
        }

        int distance = 0;
        
        // Iterate through the strings

        }
        return distance;
