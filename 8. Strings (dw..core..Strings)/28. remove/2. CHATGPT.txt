package com.example.demo.util;

public class StringUtils {

    public static String remove(String input, String toRemove) {
        if (input == null || toRemove == null) return input;
        return input.replace(toRemove, "");
    }
}

Spring Boot Controller:


package com.example.demo.controller;

import com.example.demo.util.StringUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.Collections;
import java.util.Map;

@RestController
public class RemoveController {

    @GetMapping("/remove")
    public Map<String, String> removeSubstring() {
        String input = "lazyness purity state higher-order stateful";
        String output = StringUtils.remove(input, "state");

        return Collections.singletonMap("result", output);
    }
