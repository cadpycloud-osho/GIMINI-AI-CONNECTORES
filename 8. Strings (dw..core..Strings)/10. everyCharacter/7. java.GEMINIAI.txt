    /**
     * Static method to emulate DataWeave's everyCharacter $ == " " or isNumeric($)
     * * To match the observed DataWeave behavior (which returns 'true' for the input 
     * "12 34\u00A0 56"), we must explicitly include the Non-Breaking Space (\u00A0)
     * as an acceptable character, even though the DW expression only specifies 
     * a standard space (" ").
     */
    public static boolean checkEveryCharacter(String input) {
        if (input == null || input.isEmpty()) {
            return true;
        }
        
        // Iterate over every character in the string
        for (char c : input.toCharArray()) {
            // Condition 1: character is a digit
            boolean isNumeric = Character.isDigit(c);
            
            // Condition 2: character is the standard space (\u0020)
            boolean isStandardSpace = c == ' ';
            
            // Condition 3 (The DW behavior fix): character is the Non-Breaking Space (\u00A0)
            // This is added to match the 'true' output observed in the DW engine for this specific input.
            boolean isNonBreakingSpace = c == '\u00A0';
            
            // If any character fails the combined condition, return false immediately
            if (!(isNumeric || isStandardSpace || isNonBreakingSpace)) { 
                return false;
            }
        }
        // If the loop completes, all characters passed the test
        return true;
    }
