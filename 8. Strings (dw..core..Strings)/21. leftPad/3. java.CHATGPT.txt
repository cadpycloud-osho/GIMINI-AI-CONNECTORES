import java.util.LinkedHashMap;
import java.util.Map;

public class Main {

    public static String leftPad(String input, int length) {
        if (input == null) return null;
        if (length <= input.length()) return input;

        int paddingLength = length - input.length();
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < paddingLength; i++) {
            sb.append(' ');
        }
        sb.append(input);
        return sb.toString();
    }

    public static void main(String[] args) {
        Map<String, String> result = new LinkedHashMap<>();

        result.put("a", leftPad(null, 3));
        result.put("b", leftPad("", 3));
        result.put("c", leftPad("bat", 5));
        result.put("d", leftPad("bat", 3));
        result.put("e", leftPad("bat", -1));

        System.out.println(result);
    }
}
