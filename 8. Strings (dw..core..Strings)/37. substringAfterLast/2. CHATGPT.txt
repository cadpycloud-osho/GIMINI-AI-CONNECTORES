package com.example.demo.util;

public class StringUtils {

    public static String substringAfterLastDW(String input, String delimiter) {

        // DW behavior:
        if (input == null) return null;
        if (delimiter == null) return null;
        if (delimiter.equals("")) return null; // important DW rule

        if (input.equals("")) return "";

        int index = input.lastIndexOf(delimiter);

        if (index == -1) return "";  // delimiter not found â†’ empty string

        return input.substring(index + delimiter.length());
    }
}
