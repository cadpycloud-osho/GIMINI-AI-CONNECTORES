public class CharCodeAtConverter {
    public static int charCodeAt(String input, int index) {
        if (input == null || index < 0 || index >= input.length()) {
            throw new StringIndexOutOfBoundsException("Invalid index: " + index);
        }
        return input.charAt(index);
    }
