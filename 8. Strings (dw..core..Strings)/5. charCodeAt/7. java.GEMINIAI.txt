    public static void main(String[] args) {

        String inputString = "MuleSoft";

            char charAtIndex = inputString.charAt(index);
            
            // 3. Casting the char to an int gives its ASCII/Unicode value
            int charCodeValue = (int) charAtIndex;
            
            // 4. Print the result in a format similar to the desired JSON output
            System.out.println("{ \"charCodeAt\": " + charCodeValue + " }");
            
        } else {
            System.out.println("Error: Index is out of bounds for the string.");
