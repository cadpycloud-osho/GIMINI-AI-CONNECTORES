public class SubstringUtil {

    /**
     * Replicates the dw::core::Strings::substringAfter function.
     * Returns the substring after the first occurrence of the separator.
     */
    public static String substringAfter(String str, String separator) {
        // 1. DataWeave Rule 'a': If input string is null, return null.
        if (str == null) {
            return null;
        }
        
        // 2. DataWeave Rule 'f': If separator is empty, return the original string.
        if (separator == null || separator.isEmpty()) {
            return str;
        }

        // Find the index of the first occurrence of the separator.
        int separatorIndex = str.indexOf(separator);

        // 3. DataWeave Rule 'b', 'e': If separator is not found, or input string is empty, return empty string.
        if (separatorIndex == -1) {
            return "";
        }

        // 4. Return the substring after the separator (Rules 'c', 'd').
        // Start index = index of separator + length of separator
        return str.substring(separatorIndex + separator.length());
    }
}
