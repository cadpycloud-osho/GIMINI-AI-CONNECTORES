package com.example.demo.util;

public class StringUtils {

    public static String substringAfterDW(String input, String delimiter) {

        // DW behaviour
        if (input == null) return null;

        if (delimiter == null) delimiter = "";

        // if delimiter = "" → return entire string
        if (delimiter.isEmpty()) {
            return input;
        }

        int index = input.indexOf(delimiter);

        // delimiter not found → return ""
        if (index == -1) {
            return "";
        }

        // return substring after the delimiter
        return input.substring(index + delimiter.length());
    }
}

package com.example.demo.controller;
