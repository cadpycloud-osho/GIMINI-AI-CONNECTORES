package com.example.demo.controller;

import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class IsAlphanumericController {
    
    @GetMapping(value = "/is-alphanumeric", produces = MediaType.APPLICATION_JSON_VALUE)
    public IsAlphanumericResponse isAlphanumericChecks() {
        return new IsAlphanumericResponse(
            isAlphanumeric(null),
            isAlphanumeric(""),
            isAlphanumeric("  "),
            isAlphanumeric("abc"),
            isAlphanumeric("ab c"),
            isAlphanumeric("ab2c"),
            isAlphanumeric("ab-c")
        );
    }
    
    private boolean isAlphanumeric(String input) {
        if (input == null || input.isEmpty()) {
            return false;
        }
        return input.chars().allMatch(c -> Character.isLetterOrDigit((char) c));
    }
    
    public static class IsAlphanumericResponse {
        public final boolean a, b, c, d, e, f, g;
        
        public IsAlphanumericResponse(boolean a, boolean b, boolean c, boolean d, boolean e, boolean f, boolean g) {
            this.a = a; this.b = b; this.c = c; this.d = d; this.e = e; this.f = f; this.g = g;
        }
    }
}
