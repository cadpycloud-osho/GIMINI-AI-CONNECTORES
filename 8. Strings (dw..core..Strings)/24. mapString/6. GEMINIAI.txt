import org.springframework.web.bind.annotation.RestController;
import java.util.regex.Pattern;

public class DataWeaveController {

    private final Pattern digitPattern = Pattern.compile("\\d");

     * Simulates the DataWeave transformation: 
     * Replacing every numeric character in "$234" with "~".
     *
     * @return A JSON object matching the DataWeave output structure.
     */
    @GetMapping("/transform")
    public TransformationResult transformDataWeaveString() {
        String input = "$234";
        
        // --- Core DataWeave Logic ---
        // Implementation using Regular Expression: 
        // "\\d" matches any digit. replaceAll replaces all matches with "~".
        String transformedBalance = input.replaceAll("\\d", "~");

    }

     * Inner class representing the JSON output structure.
