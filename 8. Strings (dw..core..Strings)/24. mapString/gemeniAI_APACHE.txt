import org.apache.camel.builder.RouteBuilder;
import java.util.stream.Collectors;

public class MapStringRoute extends RouteBuilder {

    @Override
    public void configure() throws Exception {
        from("direct:maskBalance")
            .process(exchange -> {
                String input = "$234";

                // Map logic: Iterate through chars, check digit, transform
                String masked = input.chars()
                    .mapToObj(c -> {
                        char character = (char) c;
                        return Character.isDigit(character) ? "~" : String.valueOf(character);
                    })
                    .collect(Collectors.joining());

                exchange.getIn().setBody(java.util.Map.of("balance", masked));
            })
            .marshal().json();
    }
}
