     * Static method to emulate DataWeave's isWhitespace(text) function.
     * Returns true if the string is empty or contains ONLY whitespace characters.
     * Returns false for null input.
     */
    public static boolean isWhitespaceString(String input) {
        // 1. DataWeave returns false for null.
        if (input == null) {
            return false;
        }

        // 2. DataWeave returns true for empty string.
        if (input.isEmpty()) {
            return true;
        }
        
        // 3. Check every character
        for (char c : input.toCharArray()) {
            // Character.isWhitespace() handles standard spaces, tabs, newlines, and Unicode whitespace.
            if (!Character.isWhitespace(c)) {

        return true;
