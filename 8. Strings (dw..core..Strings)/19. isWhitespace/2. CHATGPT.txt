package com.example.demo.util;

public class StringUtils {

    public static boolean isWhitespaceDW(String input) {
        if (input == null) return false; // DW returns false for null
        return input.trim().isEmpty(); // true for "" or spaces only
    }
}

Spring Boot Controller:
package com.example.demo.controller;

import com.example.demo.util.StringUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class WhitespaceController {

    @GetMapping("/iswhitespace")
    public Map<String, Boolean> checkWhitespace() {
        Map<String, Boolean> result = new LinkedHashMap<>();

        result.put("a", StringUtils.isWhitespaceDW(null));
        result.put("b", StringUtils.isWhitespaceDW(""));
        result.put("c", StringUtils.isWhitespaceDW("  "));
        result.put("d", StringUtils.isWhitespaceDW("abc"));
        result.put("e", StringUtils.isWhitespaceDW("ab2c"));
        result.put("f", StringUtils.isWhitespaceDW("ab-c"));

        return result;
    }
}
