package com.example.demo.util;

public class StringUtils {

    public static String wrapIfMissing(String input, String wrapper) {
        if (input == null) return null;
        if (input.isEmpty()) return wrapper;

        if (input.startsWith(wrapper) && input.endsWith(wrapper)) {
            return input;
        }

        return wrapper + input + wrapper;
    }

    // Overload for single char wrapper
    public static String wrapIfMissing(String input, char wrapper) {
        return wrapIfMissing(input, String.valueOf(wrapper));
    }
}

package com.example.demo.controller;

import com.example.demo.util.StringUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class WrapIfMissingController {

    @GetMapping("/wrapIfMissing")
    public Map<String, String> wrapIfMissingExample() {
        Map<String, String> result = new LinkedHashMap<>();

        result.put("a", StringUtils.wrapIfMissing(null, "'"));
        result.put("b", StringUtils.wrapIfMissing("", "'"));
        result.put("c", StringUtils.wrapIfMissing("ab", "x"));
        result.put("d", StringUtils.wrapIfMissing("'ab'", "'"));
        result.put("e", StringUtils.wrapIfMissing("/", '/'));
        result.put("f", StringUtils.wrapIfMissing("a/b/c", '/'));
        result.put("g", StringUtils.wrapIfMissing("/a/b/c", '/'));
        result.put("h", StringUtils.wrapIfMissing("a/b/c/", '/'));

        return result;
    }
}
