import java.util.Arrays;
import java.util.Collections;
import java.util.List;

public class Main {

    /**
     * Static method to emulate DataWeave's lines(text) function.
     * Splits a string by line breaks and keeps empty lines.
     */
    public static List<String> splitIntoLines(String input) {
        if (input == null) {
            return Collections.emptyList();
        }
        
        // Split using the regex for standard line breaks (\r?\n|\r).
        // The limit parameter set to -1 ensures that trailing empty strings (blank lines) 
        // resulting from consecutive delimiters are included in the output array.
        String[] linesArray = input.split("\\r?\\n|\\r", -1);
        
        return Arrays.asList(linesArray);
    }

    public static void main(String[] args) {
        // The input string: "hello world\n\nhere   data-weave"
        String inputString = "hello world\n\nhere\u00A0 \u00A0data-weave";
        
        List<String> result = splitIntoLines(inputString);
        
        System.out.println("DataWeave");
        System.out.println("Output");
        
        // Format the output to match the desired JSON-like structure
        System.out.print("[");
        for (int i = 0; i < result.size(); i++) {
            System.out.print("\"" + result.get(i) + "\"");
            if (i < result.size() - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}
