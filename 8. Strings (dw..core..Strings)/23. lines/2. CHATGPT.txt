package com.example.demo.util;

public class StringUtils {

    public static String[] lines(String input) {
        if (input == null) return new String[0];
        return input.split("\\r?\\n", -1); // include empty lines
    }
}

Spring Boot Controller:
package com.example.demo.controller;

import com.example.demo.util.StringUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.Arrays;
import java.util.HashMap;
import java.util.Map;

@RestController
public class LinesController {

    @GetMapping("/lines")
    public Map<String, Object> getLines() {
        Map<String, Object> response = new HashMap<>();

        String input = "hello world\n\nhere   data-weave";
        String[] result = StringUtils.lines(input);

        response.put("lines", Arrays.asList(result));

        return response;
    }
}
