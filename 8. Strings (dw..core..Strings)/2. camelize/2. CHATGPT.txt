package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

@RestController
public class CamelizeController {

    @GetMapping("/camelize")
    public Map<String, String> camelizeStrings() {
        Map<String, String> response = new HashMap<>();

        response.put("a", camelize("customer_first_name"));
        response.put("b", camelize("_name_starts_with_underscore"));

        return response;
    }

    private String camelize(String input) {
        String[] parts = input.split("_");
        StringBuilder result = new StringBuilder();

        for (int i = 0; i < parts.length; i++) {
            if (parts[i].isEmpty()) continue; // Skip leading underscore
            if (i == 0) {
                result.append(parts[i].toLowerCase());
            } else {
                result.append(parts[i].substring(0, 1).toUpperCase())
                      .append(parts[i].substring(1).toLowerCase());
            }
        }
        return result.toString();
    }
}
