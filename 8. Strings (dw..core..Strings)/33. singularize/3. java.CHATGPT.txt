import java.util.Map;

public class Main {

    public static String singularize(String input) {
        if (input == null || input.isEmpty()) return input;

        if (input.endsWith("es") && (input.endsWith("xes") || input.endsWith("ses") || input.endsWith("zes") ||
                input.endsWith("ches") || input.endsWith("shes"))) {
            return input.substring(0, input.length() - 2);
        }

        if (input.endsWith("ies")) {
            return input.substring(0, input.length() - 3) + "y";
        }

        if (input.endsWith("s") && input.length() > 1) {
            return input.substring(0, input.length() - 1);
        }

        return input;
    }

    public static void main(String[] args) {
        Map<String, String> result = Map.of(
                "singularize", singularize("boxes")
        );

        System.out.println(result);
    }
}
