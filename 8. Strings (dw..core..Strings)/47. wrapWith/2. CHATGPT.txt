package com.example.demo.util;

public class StringUtils {

    public static String wrapWithDW(String input, String wrapper) {
        if (input == null) return null;       // DW behavior
        if (wrapper == null) wrapper = "";    // Safety

        return wrapper + input + wrapper;
    }
}


package com.example.demo.controller;

import com.example.demo.util.StringUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class WrapWithController {

    @GetMapping("/wrapwith")
    public Map<String, String> wrapWithExample() {
        Map<String, String> result = new LinkedHashMap<>();

        result.put("a", StringUtils.wrapWithDW(null, "'"));
        result.put("b", StringUtils.wrapWithDW("", "'"));
        result.put("c", StringUtils.wrapWithDW("ab", "x"));
        result.put("d", StringUtils.wrapWithDW("'ab'", "'"));
        result.put("e", StringUtils.wrapWithDW("ab", "'"));

        return result;
    }
}
