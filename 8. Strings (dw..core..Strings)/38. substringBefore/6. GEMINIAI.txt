public class SubstringUtil {

    /**
     * Replicates the dw::core::Strings::substringBefore function.
     * Returns the substring before the first occurrence of the separator.
     */
    public static String substringBefore(String str, String separator) {
        // 1. DataWeave Rule 'a': If input string is null, return null.
        if (str == null) {
            return null;
        }
        
        // 2. DataWeave Rule 'f': If separator is empty, return empty string.
        if (separator == null || separator.isEmpty()) {
            return "";
        }

        // Find the index of the first occurrence of the separator.
        int separatorIndex = str.indexOf(separator);

        // 3. DataWeave Rule 'b', 'e': If separator is not found, return empty string.
        if (separatorIndex == -1) {
            return "";
        }

        // 4. Return the substring before the separator (Rules 'c', 'd').
        // Java's substring(0, index) is exclusive of the end index.
        return str.substring(0, separatorIndex);
    }
}
