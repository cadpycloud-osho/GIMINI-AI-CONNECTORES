package com.example.demo.util;

public class StringUtils {

    public static String substringBeforeLastDW(String input, String delimiter) {
        if (input == null) return null;
        if (delimiter == null) delimiter = "";

        if (delimiter.equals("")) {
            if (input.length() <= 1) return "";
            return input.substring(0, input.length() - 1);
        }

        int index = input.lastIndexOf(delimiter);
        if (index == -1) return "";
        return input.substring(0, index);
    }
}

package com.example.demo.controller;

import com.example.demo.util.StringUtils;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class SubstringBeforeLastController {

    @GetMapping("/substringbeforelast")
    public Map<String, String> substringBeforeLastExample() {
        Map<String, String> result = new LinkedHashMap<>();

        result.put("a", StringUtils.substringBeforeLastDW(null, "'"));
        result.put("b", StringUtils.substringBeforeLastDW("", "-"));
        result.put("c", StringUtils.substringBeforeLastDW("abc", "b"));
        result.put("d", StringUtils.substringBeforeLastDW("abcba", "b"));
        result.put("e", StringUtils.substringBeforeLastDW("abc", "d"));
        result.put("f", StringUtils.substringBeforeLastDW("abc", ""));

        return result;
    }
