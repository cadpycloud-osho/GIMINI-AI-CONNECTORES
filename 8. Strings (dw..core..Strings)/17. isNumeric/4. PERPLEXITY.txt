package com.example.demo.controller;

import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class IsNumericController {
    
    @GetMapping(value = "/is-numeric", produces = MediaType.APPLICATION_JSON_VALUE)
    public IsNumericResponse isNumericChecks() {
        return new IsNumericResponse(
            isNumeric(null),
            isNumeric(""),
            isNumeric("  "),
            isNumeric("123"),
            isNumeric("резреирей"),
            isNumeric("12 3"),
            isNumeric("ab2c"),
            isNumeric("12-3"),
            isNumeric("12.3"),
            isNumeric("-123"),
            isNumeric("+123")
        );
    }
    
    private boolean isNumeric(String input) {
        if (input == null || input.isEmpty()) {
            return false;
        }
        return input.chars().allMatch(Character::isDigit);
    }
    
    public static class IsNumericResponse {
        public final boolean a, b, c, d, e, f, g, h, i, j, k;
        
        public IsNumericResponse(boolean a, boolean b, boolean c, boolean d, boolean e, 
                               boolean f, boolean g, boolean h, boolean i, boolean j, boolean k) {
            this.a = a; this.b = b; this.c = c; this.d = d; this.e = e;
            this.f = f; this.g = g; this.h = h; this.i = i; this.j = j; this.k = k;
        }
    }
}
