public class Main {

    /**
     * Static method to emulate DataWeave's isNumeric(text) function.
     * Returns true if the string is NOT null, NOT empty, and contains ONLY numeric digits.
     */
    public static boolean isNumericString(String input) {
        // 1. DataWeave returns false for null and empty strings.
        if (input == null || input.isEmpty()) {
            return false;
        }

        // 2. Check every character
        for (char c : input.toCharArray()) {
            // Character.isDigit() handles standard digits and many Unicode digit forms.
            if (!Character.isDigit(c)) {
                return false;
            }
        }
        // If the loop completes, all characters are numeric digits.
        return true;
    }

    public static void main(String[] args) {
        // Testing various inputs:
        boolean a = isNumericString(null);
        boolean b = isNumericString("");
        boolean c = isNumericString("  "); // Contains only spaces, not digits -> false
        boolean d = isNumericString("123");

        boolean g = isNumericString("ab2c"); // Contains letters -> false
        boolean h = isNumericString("12-3"); // Contains hyphen -> false
        boolean i = isNumericString("12.3"); // Contains decimal point -> false
