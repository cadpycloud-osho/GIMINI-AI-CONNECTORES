import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import java.util.LinkedHashMap;
import java.util.Map;

@RestController
public class IsAlphaController {

    /**
     * Custom method to emulate DataWeave's isAlpha(text) function.
     * Returns true if the string is NOT null, NOT empty, and contains ONLY alphabetic characters.
     */
    private boolean isAlphaString(String input) {
        // DataWeave's isAlpha returns false for null and empty strings.
        if (input == null || input.isEmpty()) {
            return false;
        }

        // Check if ALL characters in the string are letters.
        return input.chars() // Get an IntStream of character codes
                    .allMatch(Character::isLetter); // Check if every code is a letter
    }

    /**
     * Endpoint equivalent to the DataWeave expression block.
     */
    @GetMapping("/check-is-alpha")
    public Map<String, Boolean> checkIsAlphaExamples() {
        // Use LinkedHashMap to preserve the output order
        Map<String, Boolean> result = new LinkedHashMap<>();

        // Note: The non-breaking space ("  ") in 'c' fails Character.isLetter()
        result.put("a", isAlphaString(null));
        result.put("b", isAlphaString(""));
        result.put("c", isAlphaString("  ")); // Contains space/non-breaking space
        result.put("d", isAlphaString("abc"));
        result.put("e", isAlphaString("ab2c")); // Contains digit
        result.put("f", isAlphaString("ab-c")); // Contains hyphen
        
        return result;
    }
}
