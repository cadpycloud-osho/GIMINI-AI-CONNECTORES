package com.example.demo.controller;

import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class IsAlphaController {
    
    @GetMapping(value = "/is-alpha", produces = MediaType.APPLICATION_JSON_VALUE)
    public IsAlphaResponse isAlphaChecks() {
        return new IsAlphaResponse(
            isAlpha(null),
            isAlpha(""),
            isAlpha("  "),
            isAlpha("abc"),
            isAlpha("ab2c"),
            isAlpha("ab-c")
        );
    }
    
    private boolean isAlpha(String input) {
        if (input == null || input.isEmpty()) {
            return false;
        }
        return input.chars().allMatch(Character::isLetter);
    }
    
    public static class IsAlphaResponse {
        public final boolean a, b, c, d, e, f;
        
        public IsAlphaResponse(boolean a, boolean b, boolean c, boolean d, boolean e, boolean f) {
            this.a = a; this.b = b; this.c = c; this.d = d; this.e = e; this.f = f;
        }
    }
}
