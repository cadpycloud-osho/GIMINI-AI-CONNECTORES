package com.example.demo.controller;

import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class IsUpperCaseController {
    
    @GetMapping(value = "/is-uppercase", produces = MediaType.APPLICATION_JSON_VALUE)
    public IsUpperCaseResponse isUpperCaseChecks() {
        return new IsUpperCaseResponse(
            isUpperCase(null),
            isUpperCase(""),
            isUpperCase("  "),
            isUpperCase("ABC"),
            isUpperCase("aBC"),
            isUpperCase("A C"),
            isUpperCase("A1C"),
            isUpperCase("A/C")
        );
    }
    
    private boolean isUpperCase(String input) {
        if (input == null || input.isEmpty()) {
            return false;
        }
        return input.chars().allMatch(Character::isUpperCase);
    }
    
    public static class IsUpperCaseResponse {
        public final boolean a, b, c, d, e, f, g, h;
        
        public IsUpperCaseResponse(boolean a, boolean b, boolean c, boolean d, boolean e, 
                                 boolean f, boolean g, boolean h) {
            this.a = a; this.b = b; this.c = c; this.d = d; this.e = e;
            this.f = f; this.g = g; this.h = h;
        }
    }
}
