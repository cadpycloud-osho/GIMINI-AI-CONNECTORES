import java.util.*;

public class IsUpperCaseExample {

    public static void main(String[] args) {

        Map<String, Boolean> output = new LinkedHashMap<>();

        output.put("a", isUpperCase(null));
        output.put("b", isUpperCase(""));
        output.put("c", isUpperCase("  "));
        output.put("d", isUpperCase("ABC"));
        output.put("e", isUpperCase("aBC"));
        output.put("f", isUpperCase("A C"));
        output.put("g", isUpperCase("A1C"));
        output.put("h", isUpperCase("A/C"));

        System.out.println(output);
    }

    private static boolean isUpperCase(String str) {
        if (str == null || str.isEmpty()) return false;

        for (char c : str.toCharArray()) {
            if (!Character.isLetter(c)) return false;
            if (!Character.isUpperCase(c)) return false;
        }
        return true;
    }
}
