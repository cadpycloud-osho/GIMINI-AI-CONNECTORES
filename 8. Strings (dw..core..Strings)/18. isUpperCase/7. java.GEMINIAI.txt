public class Main {

    /**
     * Static method to emulate DataWeave's isUpperCase(text) function.
     * Returns true if the string is NOT null, NOT empty, and contains ONLY uppercase letters.
     */
    public static boolean isUpperCaseString(String input) {
        // 1. DataWeave returns false for null and empty strings.
        if (input == null || input.isEmpty()) {
            return false;
        }

        // 2. Check every character
        for (char c : input.toCharArray()) {
            // Must be a letter AND must be uppercase. 
            // This ensures we reject spaces, digits, and punctuation.
            if (!Character.isLetter(c) || !Character.isUpperCase(c)) {
                return false;
            }
        }
        // If the loop completes, all characters are uppercase letters.
        return true;
    }
