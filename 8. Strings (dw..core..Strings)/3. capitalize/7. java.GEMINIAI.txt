public class Main {

    /**
     * Implements the logic equivalent to DataWeave's capitalize (Title Case).
     */
    public static String capitalize(String input) {
        if (input == null || input.isEmpty()) {
            return input;
        }

        // 1. Normalize separators: Convert underscores to spaces and detect camelCase boundaries
        String normalized = input
            .replaceAll("_", " ") 
            .replaceAll("([a-z])([A-Z])", "$1 $2"); // Insert space for camelCase

        // 2. Split the string into words using any whitespace as a delimiter
        String[] words = normalized.split("\\s+");
        StringBuilder result = new StringBuilder();

        for (int i = 0; i < words.length; i++) {
            String word = words[i];
            if (word.isEmpty()) {
                continue;
            }

            // Capitalize the first letter
            result.append(Character.toUpperCase(word.charAt(0)));
            
            // Append the rest of the word
            if (word.length() > 1) {
                // Ensure the rest of the word is maintained in its original case (as DW does)
                result.append(word.substring(1));
            }
            
            // Append a space if it's not the last word
            if (i < words.length - 1) {
                result.append(" ");
            }
        }

        return result.toString().trim();
    }

    public static void main(String[] args) {
        // Source cases:
        String a = capitalize("customer");
        String b = capitalize("customer_first_name");
        String c = capitalize("customer NAME");
        String d = capitalize("customerName");
        String e = capitalize("a*s_b’s");

        // --- Final JSON Output ---
        System.out.println("{");
        System.out.println("  \"a\": \"" + a + "\",");
        System.out.println("  \"b\": \"" + b + "\",");
        System.out.println("  \"c\": \"" + c + "\",");
        System.out.println("  \"d\": \"" + d + "\",");
        System.out.println("  \"e\": \"" + e + "\"");
        System.out.println("}");
    }
}
