package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.time.Instant;
import java.time.Duration;
import java.util.HashMap;
import java.util.Map;

@RestController
public class DateTimeController {

    @GetMapping("/periods")
    public Map<String, Object> getTimePeriods() {
        Map<String, Object> response = new HashMap<>();

        Instant inputTime = Instant.parse("2020-10-05T20:22:34.385Z");

        // Operations
        Instant nextSecond = inputTime.plusSeconds(1);
        Instant previousSecond = inputTime.minusSeconds(1);
        Duration decimalInputPeriod = Duration.ofMillis((long) (4.555 * 1000)); 
        Duration wholeNumberInputPeriod = Duration.ofSeconds(4);
        long addNegativeValue = Duration.ofSeconds(-1).plusSeconds(2).getSeconds();

        // Create response JSON-like map
        response.put("nextSecond", nextSecond.toString());
        response.put("previousSecond", previousSecond.toString());
        response.put("decimalInputPeriod", decimalInputPeriod.toString());
        response.put("wholeNumberInputPeriod", wholeNumberInputPeriod.toString());
        response.put("addNegativeValue", addNegativeValue);

        return response;
    }
}
