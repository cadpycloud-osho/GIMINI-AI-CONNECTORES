package com.example.demo.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import java.time.ZonedDateTime;
import java.time.Duration;
import java.time.format.DateTimeFormatter;
import java.util.HashMap;
import java.util.Map;

@RestController
public class PeriodController {

    @GetMapping("/periods")
    public Map<String, Object> getPeriods() {
        Map<String, Object> response = new HashMap<>();
        ZonedDateTime inputDateTime = ZonedDateTime.parse("2020-10-05T20:22:34.385Z");

        ZonedDateTime nextMinute = inputDateTime.plusMinutes(1);
        ZonedDateTime previousMinute = inputDateTime.minusMinutes(1);

        Duration decimalDuration = Duration.ofMillis((long) (4.555 * 60 * 1000));
        Duration wholeDuration = Duration.ofMinutes(4);
        Duration addNegativeDuration = Duration.ofMinutes(-1).plusMinutes(2);

        DateTimeFormatter formatter = DateTimeFormatter.ISO_ZONED_DATE_TIME;

        response.put("nextMinute", formatter.format(nextMinute));
        response.put("previousMinute", formatter.format(previousMinute));
        response.put("decimalInputPeriod", decimalDuration.toString());
        response.put("wholeNumberInputPeriod", wholeDuration.toString());
        response.put("addNegativeValue", addNegativeDuration.toMinutes());

        return response;
    }
}
